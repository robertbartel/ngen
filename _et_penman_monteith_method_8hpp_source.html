78//#####################################################################################*
79// subroutine to calculate latent heat flux (lambda*et) *
80// using the Penman-Monteith equation as described by FAO *
81// see: http://www.fao.org/3/X0490E/x0490e06.htm#aerodynamic%20resistance%20(ra) *
82// After Allen et al. ASCE Reference ET calculation method *
83// F.L. Ogden, NOAA National Weather Service, 2020 *
84//#####################################################################################*
85extern double penman_monteith_et_calculation
86(
87 double delta,
88 double gamma,
89 double moist_air_density_kg_per_m3,
90 double vapor_pressure_deficit_Pa,
91 evapotranspiration_params *et_params,
92 evapotranspiration_forcing *et_forcing
93)
94
95{
96 // local varibles
97 double numerator;
98 double denominator;
99 double aerodynamic_resistance_s_per_m;
100
101 // this method requires more calculations
102 if(is_fabs_less_than_eps(et_params->vegetation_height_m,1.0e-06)==TRUE)
103 {
104 // the vegetation height was not specified. TODO should warn??
105 fprintf(stderr,"WARNING: Vegetation height not specified in the Penman-Monteith routine. Using 0.5m.\n");
106 et_params->vegetation_height_m=0.5; // use a reasonable assumed value
107 }
108
109 // use approximations from UN FAO: http://www.fao.org/3/X0490E/x0490e06.htm#aerodynamic%20resistance%20(ra)
110 et_params->zero_plane_displacement_height_m=2.0/3.0*et_params->vegetation_height_m;
111 et_params->momentum_transfer_roughness_length_m=0.123*et_params->vegetation_height_m;
112 et_params->heat_transfer_roughness_length_m=0.1*et_params->vegetation_height_m;
113
114 aerodynamic_resistance_s_per_m= calculate_aerodynamic_resistance
115 ( et_params->wind_speed_measurement_height_m,
116 et_params->humidity_measurement_height_m,
117 et_params->zero_plane_displacement_height_m,
118 et_params->momentum_transfer_roughness_length_m,
119 et_params->heat_transfer_roughness_length_m,
120 et_forcing->wind_speed_m_per_s);
121
122 // all the ingredients have been prepared. Make Penman-Monteith soufle...
123 // from: http://www.fao.org/3/X0490E/x0490e06.htm#aerodynamic%20resistance%20(ra)
124
125 numerator = delta* (et_forcing->net_radiation_W_per_sq_m - et_forcing->ground_heat_flux_W_per_sq_m) +
126 moist_air_density_kg_per_m3 * CP *
127 vapor_pressure_deficit_Pa/aerodynamic_resistance_s_per_m;
128
129 denominator = delta + gamma * (1.0+et_forcing->canopy_resistance_sec_per_m/aerodynamic_resistance_s_per_m);
130
131 return(numerator/denominator); // Latent heat flux in Watts per sq. m., or J per (s m2)
132}
