104 static int mass_check(const hymod_params& params, const hymod_state& current_state, const hymod_state& next_state, const hymod_fluxes& calculated_fluxes, double timestep_seconds)
105 {
106 // initalize both mass values from current and next states storage

108 double final_mass_meters = next_state.storage_meters + next_state.groundwater_storage_meters;
109
110 // add the masses of the reservoirs before and after the time step
111 for ( int i = 0; i < params.n; ++i)
112 {
113 inital_mass_meters += current_state.Sr[i];
114 final_mass_meters += next_state.Sr[i];
115 }
116 // increase the inital mass by input value, NJF run function above adds input_flux to state before it is passed here
117 //inital_mass_meters += input_flux_meters;
118
119 // increase final mass by calculated fluxes
120 final_mass_meters += (calculated_fluxes.et_loss_meters + calculated_fluxes.runoff_meters_per_second * timestep_seconds + calculated_fluxes.slow_flow_meters_per_second * timestep_seconds);
121
122 if ( fabs(inital_mass_meters - final_mass_meters) > 0.000001 )
123 {
124 return MASS_BALANCE_ERROR;
