38 NetCDFPerFeatureDataProvider(std::string input_path, utils::StreamHandler log_s) : log_stream(log_s)
39 {
40 //open the file
41 nc_file = std::make_shared<NcFile>(input_path, NcFile::read);
42
43 //get the listing of all variables
44 auto var_set = nc_file->getVars();
45
46 // copy the variables names into the vector for easy use
47 std::for_each(var_set.begin(), var_set.end(), [&](const auto& element)
48 {
49 variable_names.push_back(element.first);
50 });
51
52 // read the variable ids
53 auto ids = nc_file->getVar("ids");
54 auto id_dim_count = ids.getDimCount();
55
56 // some sanity checks
57 if ( id_dim_count > 1)
58 {
59 throw std::runtime_error("Provided NetCDF file has an \"ids\" variable with more than 1 dimension");
60 }
61
62 auto id_dim = ids.getDim(0);
63
64 if (id_dim.isNull() )
65 {
66 throw std::runtime_error("Provided NetCDF file has a NuLL dimension for variable \"ids\"");
67 }
68
69 auto num_ids = id_dim.getSize();
70
71 // allocate an array of character pointers
72 std::vector< char* > string_buffers(num_ids);
73
74 // read the id strings
75 ids.getVar(&string_buffers[0]);
76
77 // initalize the map of catchment-name to offset location and free the strings allocated by the C library
78 size_t loc = 0;
79 for_each( string_buffers.begin(), string_buffers.end(), [&](char* str)
80 {
81 loc_ids.push_back(str);
82 id_pos[str] = loc++;
83 });
84
85 // correct string release
86 ids.freeString(num_ids,&string_buffers[0]);
87
88 // now get the size of the time dimension
89 auto num_times = nc_file->getDim("time").getSize();
90
91 // allocate storage for the raw time array
92 std::vector<double> raw_time(num_times);
93
94 // get the time variable
95 auto time_var = nc_file->getVar("Time");
96
97 // read from the first catchment row to get the recorded times
98 std::vector<size_t> start;
99 start.push_back(0);
100 start.push_back(0);
101 std::vector<size_t> count;
102 count.push_back(1);
103 count.push_back(num_times);
104 time_var.getVar(start, count, &raw_time[0]);
105
106 // read the meta data to get the time_unit
107 auto time_unit_att = time_var.getAtt("units");
108
109 // if time att is not encoded
110 // TODO determine how this should be handled
111 std::string time_unit_str;
112 double time_scale_factor;
113
114 if ( time_unit_att.isNull() )
115 {
116 time_unit = TIME_SECONDS;
117 time_scale_factor = 1;
118 }
119 else
120 {
121 time_unit_att.getValues(time_unit_str);
122 }
123
124 // set time unit and scale factor
125 if ( time_unit_str == "h" || time_unit_str == "hours ")
126 {
127 time_unit = TIME_HOURS;
128 time_scale_factor = 3600;
129 }
130 else if ( time_unit_str == "m" || time_unit_str == "minutes" )
131 {
132 time_unit = TIME_MINUETS;
133 time_scale_factor = 60;
134 }
135 else if ( time_unit_str == "s" || time_unit_str == "seconds" )
136 {
137 time_unit = TIME_SECONDS;
138 time_scale_factor = 1;
139 }
140 else if ( time_unit_str == "ms" || time_unit_str == "milliseconds" )
141 {
142 time_unit = TIME_MILLISECONDS;
143 time_scale_factor = .001;
144 }
145 else if ( time_unit_str == "us" || time_unit_str == "microseconds" )
146 {
147 time_unit = TIME_MICROSECONDS;
148 time_scale_factor = .000001;
149 }
150 else if ( time_unit_str == "ns" || time_unit_str == "nanoseconds" )
151 {
152 time_unit = TIME_SECONDS;
153 time_scale_factor = .000000001;
154 }
155
156 // read the meta data to get the epoc start
157 auto epoc_att = time_var.getAtt("epoc_start");
158 std::string epoc_start_str;
159
160 if ( epoc_att.isNull() )
161 {
162 epoc_start_str = "01/01/1970 00:00:00";
163 log_stream << "Warning using defualt epoc string\n";
164 }
165 else
166 {
167 epoc_att.getValues(epoc_start_str);
168 }
169
170 std::tm tm{};
171 std::stringstream s(epoc_start_str);
172 s >> std::get_time(&tm, "%F %R");
173 std::time_t epoc_start_time = mktime(&tm);
174
175 // scale the time to account for time units and epoc_start
176 // TODO make sure this happens with a FMA instruction
177 time_vals.resize(raw_time.size());
178 std::transform(raw_time.begin(), raw_time.end(), time_vals.begin(),
179 [&](const auto& n){return n * time_scale_factor + epoc_start_time; });
180
181
182 // determine the stride of the time array
183 time_stride = time_vals[1] - time_vals[0];
184
185 #ifndef NCEP_OPERATIONS
186 // verify the time stride
187 for( size_t i = 1; i < time_vals.size() -1; ++i)
188 {
189 double tinterval = time_vals[i+1] - time_vals[i];
190
191 if ( tinterval - time_stride > 0.000001)
192 {
193 log_stream << "Error: Time intervals are not constant in forcing file\n";
194
195 throw std::runtime_error("Time intervals in forcing file are not constant");
196 }
197 }
198 #endif
199
200 // determine start_time and stop_time;
201 start_time = time_vals[0];
202 stop_time = time_vals.back() + time_stride;
203
204
205 }

280 double get_value(const CatchmentAggrDataSelector& selector, ReSampleMethod m) override
281 {
282 auto init_time = selector.get_init_time();
283 auto stop_time = init_time + selector.get_duration_secs();
284
285 auto idx1 = get_ts_index_for_time(init_time);
286 auto idx2 = get_ts_index_for_time(stop_time);
287
288 auto stride = idx2 - idx1;
289
290 std::vector<std::size_t> start, count;
291
292 auto cat_pos = id_pos[selector.get_id()];
293
294 start.push_back(cat_pos);
295 start.push_back(idx1);
296
297 count.push_back(1);
298
299 double t1 = time_vals[idx1];
300 double t2 = time_vals[idx2];
301
302 double rvalue = 0.0;
303
304 auto ncvar = nc_file->getVar(selector.get_variable_name());
305
306 std::string native_units;
307
308 try
309 {
310 auto units_att = ncvar.getAtt("units");
311 if ( units_att.isNull() )
312 {
313 native_units = "unknown";
314 }
315 else
316 {
317 units_att.getValues(native_units);
318 }
319 }
320 catch(...)
321 {
322 native_units = "unknown";
323 }
324
325 auto read_len = idx2 - idx1 + 1;
326 count.push_back(read_len);
327
328 std::vector<double> raw_values;
329 raw_values.resize(read_len);
330
331 ncvar.getVar(start,count,&raw_values[0]);
332
333 rvalue = 0.0;
334
335 double a , b = 0.0;
336
337 a = 1.0 - ( (t1 - init_time) / time_stride );
338 rvalue += (a * raw_values[0]);
339
340 for( size_t i = 1; i < raw_values.size() -1; ++i )
341 {
342 rvalue += raw_values[i];
343 }
344
345 if ( raw_values.size() > 1) // likewise the last data value may not be fully in the window
346 {
347 b = (stop_time - t2) / time_stride;
348 rvalue += (b * raw_values.back() );
349 }
350
351 // account for the resampling methods
352 switch(m)
353 {
354 case SUM: // we allready have the sum so do nothing
355 ;
356 break;
357
358 case MEAN:
359 {
360 // This is getting a length weighted mean
361 // the data values where allready scaled for where there was only partial use of a data value
362 // so we just need to do a final scale to account for the differnce between time_stride and duration_s
363
364 double scale_factor = (selector.get_duration_secs() > time_stride ) ? (time_stride / selector.get_duration_secs()) : (1.0 / (a + b));
365 rvalue *= scale_factor;
366 }
367 break;
368
369 default:
370 ;
371 }
372
373 try
374 {
375 return UnitsHelper::get_converted_value(native_units, rvalue, selector.get_output_units());
376 }
377 catch (const std::runtime_error& e)
378 {
379 std::cerr<<"Unit conversion error: "<<std::endl<<e.what()<<std::endl<<"Returning unconverted value!"<<std::endl;
380 return rvalue;
381 }
382
383 return rvalue;
384 }
