65 NetCDFPerFeatureDataProvider(std::string input_path, utils::StreamHandler log_s) : log_stream(log_s), value_cache(20)
66 {
67 //size_t sizep = 1073741824, nelemsp = 202481;

71 //open the file
72 nc_file = std::make_shared<NcFile>(input_path, NcFile::read);
73
74 //nc_get_chunk_cache(&sizep, &nelemsp, &preemptionp);

77 //get the listing of all variables
78 auto var_set = nc_file->getVars();
79
80 // populate the ncvar and units caches...
81 std::for_each(var_set.begin(), var_set.end(), [&](const auto& element)
82 {
83 std::string var_name = element.first;
84 auto ncvar = nc_file->getVar(var_name);
85 variable_names.push_back(var_name);
86 ncvar_cache.emplace(var_name,ncvar);
87
88 std::string native_units;
89 try
90 {
91 auto units_att = ncvar.getAtt("units");
92 if ( units_att.isNull() )
93 {
94 native_units = "";
95 }
96 else
97 {
98 units_att.getValues(native_units);
99 }
100 }
101 catch(...)
102 {
103 native_units = "";
104 }
105
106 auto wkf = data_access::WellKnownFields.find(var_name);
107 if(wkf != data_access::WellKnownFields.end()){
108 native_units = native_units.empty() ? std::get<1>(wkf->second) : native_units;
109 std::string can_name = std::get<0>(wkf->second); // the CSDMS name
110 variable_names.push_back(can_name);
111 ncvar_cache.emplace(can_name,ncvar);
112 units_cache[can_name] = native_units;
113 }
114
115 units_cache[var_name] = native_units;
116 });
117
118 // read the variable ids
119 auto ids = nc_file->getVar("ids");
120 auto id_dim_count = ids.getDimCount();
121
122 // some sanity checks
123 if ( id_dim_count > 1)
124 {
125 throw std::runtime_error("Provided NetCDF file has an \"ids\" variable with more than 1 dimension");
126 }
127
128 auto id_dim = ids.getDim(0);
129
130 if (id_dim.isNull() )
131 {
132 throw std::runtime_error("Provided NetCDF file has a NuLL dimension for variable \"ids\"");
133 }
134
135 auto num_ids = id_dim.getSize();
136
137 //TODO: split into smaller slices if num_ids is large.
138 cache_slice_c_size = num_ids;
139
140 // allocate an array of character pointers
141 std::vector< char* > string_buffers(num_ids);
142
143 // read the id strings
144 ids.getVar(&string_buffers[0]);
145
146 // initalize the map of catchment-name to offset location and free the strings allocated by the C library
147 size_t loc = 0;
148 for_each( string_buffers.begin(), string_buffers.end(), [&](char* str)
149 {
150 loc_ids.push_back(str);
151 id_pos[str] = loc++;
152 });
153
154 // correct string release
155 ids.freeString(num_ids,&string_buffers[0]);
156
157 // now get the size of the time dimension
158 auto num_times = nc_file->getDim("time").getSize();
159
160 // allocate storage for the raw time array
161 std::vector<double> raw_time(num_times);
162
163 // get the time variable
164 auto time_var = nc_file->getVar("Time");
165
166 // read from the first catchment row to get the recorded times
167 std::vector<size_t> start;
168 start.push_back(0);
169 start.push_back(0);
170 std::vector<size_t> count;
171 count.push_back(1);
172 count.push_back(num_times);
173 time_var.getVar(start, count, &raw_time[0]);
174
175 // read the meta data to get the time_unit
176 // if absent, assume seconds
177 double time_scale_factor = 1;
178 time_unit = TIME_SECONDS;
179 try {
180 auto time_unit_att = time_var.getAtt("units");
181
182 // if time att is not encoded
183 // TODO determine how this should be handled
184 std::string time_unit_str;
185
186 if ( time_unit_att.isNull() )
187 {
188 log_stream << "Warning using defualt time units\n";
189 }
190 else
191 {
192 time_unit_att.getValues(time_unit_str);
193 }
194
195 // set time unit and scale factor
196 if ( time_unit_str == "h" || time_unit_str == "hours")
197 {
198 time_unit = TIME_HOURS;
199 time_scale_factor = 3600;
200 }
201 else if ( time_unit_str == "m" || time_unit_str == "minutes" )
202 {
203 time_unit = TIME_MINUTES;
204 time_scale_factor = 60;
205 }
206 else if ( time_unit_str == "s" || time_unit_str == "seconds" )
207 {
208 time_unit = TIME_SECONDS;
209 time_scale_factor = 1;
210 }
211 else if ( time_unit_str == "ms" || time_unit_str == "milliseconds" )
212 {
213 time_unit = TIME_MILLISECONDS;
214 time_scale_factor = .001;
215 }
216 else if ( time_unit_str == "us" || time_unit_str == "microseconds" )
217 {
218 time_unit = TIME_MICROSECONDS;
219 time_scale_factor = .000001;
220 }
221 else if ( time_unit_str == "ns" || time_unit_str == "nanoseconds" )
222 {
223 time_unit = TIME_NANOSECONDS;
224 time_scale_factor = .000000001;
225 }
226 else {
227 log_stream << "Warning using defualt time units\n";
228 }
229 }
230 catch(const netCDF::exceptions::NcException& e){
231 std::cerr<<e.what()<<std::endl;
232 log_stream << "Warning using defualt time units\n";
233 }
234 assert(time_scale_factor != 0); // This should not happen.
235
236 std::string epoch_start_str = "01/01/1970 00:00:00";
237 try {
238 // read the meta data to get the epoc start
239 auto epoch_att = time_var.getAtt("epoch_start");
240
241 if ( epoch_att.isNull() )
242 {
243 log_stream << "Warning using defualt epoc string\n";
244 }
245 else
246 {
247 epoch_att.getValues(epoch_start_str);
248 }
249 }
250 catch(const netCDF::exceptions::NcException& e) {
251 std::cerr<<e.what()<<std::endl;
252 log_stream << "Warning using defualt epoc string\n";
253 }
254
255 std::tm tm{};
256 std::stringstream s(epoch_start_str);
257 s >> std::get_time(&tm, "%D %T");
258 std::time_t epoch_start_time = mktime(&tm);
259
260 // scale the time to account for time units and epoch_start
261 // TODO make sure this happens with a FMA instruction
262 time_vals.resize(raw_time.size());
263 std::transform(raw_time.begin(), raw_time.end(), time_vals.begin(),
264 [&](const auto& n){return n * time_scale_factor + epoch_start_time; });
265
266
267 // determine the stride of the time array
268 time_stride = time_vals[1] - time_vals[0];
269
270 #ifndef NCEP_OPERATIONS
271 // verify the time stride
272 for( size_t i = 1; i < time_vals.size() -1; ++i)
273 {
274 double tinterval = time_vals[i+1] - time_vals[i];
275
276 if ( tinterval - time_stride > 0.000001)
277 {
278 log_stream << "Error: Time intervals are not constant in forcing file\n";
279
280 throw std::runtime_error("Time intervals in forcing file are not constant");
281 }
282 }
283 #endif
284
285 // determine start_time and stop_time;
286 start_time = time_vals[0];
287 stop_time = time_vals.back() + time_stride;
288

339 size_t get_ts_index_for_time(const time_t &epoch_time)
340 {
341 if (start_time <= epoch_time && epoch_time < stop_time)
342 {
343 double offset = epoch_time - start_time;
344 offset /= time_stride;

365 double get_value(const CatchmentAggrDataSelector& selector, ReSampleMethod m) override
366 {
367 auto init_time = selector.get_init_time();
368 auto stop_time = init_time + selector.get_duration_secs(); // scope hiding! BAD JUJU!
369
370 size_t idx1 = get_ts_index_for_time(init_time);
371 size_t idx2;
372 try {
373 idx2 = get_ts_index_for_time(stop_time-1); // Don't include next timestep when duration % timestep = 0
374 }
375 catch(const std::out_of_range &e){
376 idx2 = get_ts_index_for_time(this->stop_time-1); //to the edge
377 }
378
379 auto stride = idx2 - idx1;
380
381 std::vector<std::size_t> start, count;
382
383 auto cat_pos = id_pos[selector.get_id()];
384
385
386
387 double t1 = time_vals[idx1];
388 double t2 = time_vals[idx2];
389
390 double rvalue = 0.0;
391
392 auto ncvar = get_ncvar(selector.get_variable_name());
393
394 std::string native_units = get_ncvar_units(selector.get_variable_name());
395
396 auto read_len = idx2 - idx1 + 1;
397
398 std::vector<double> raw_values;
399 raw_values.resize(read_len);
400
401 //TODO: Currently assuming a whole variable cache slice across all catchments for a single timestep...but some stuff here to support otherwise.
402 size_t cache_slices_t_n = read_len / cache_slice_t_size; // Integer division!
403 // For reference: https://stackoverflow.com/a/72030286
404 for( size_t i = 0; i < cache_slices_t_n; i++ ) {
405 std::shared_ptr<std::vector<double>> cached;
406 int cache_t_idx = (idx1 - (idx1 % cache_slice_t_size) + i);
407 std::string key = ncvar.getName() + "|" + std::to_string(cache_t_idx);
408 if(value_cache.contains(key)){
409 cached = value_cache.get(key).get();
410 } else {
411 cached = std::make_shared<std::vector<double>>(cache_slice_c_size * cache_slice_t_size);
412 start.clear();
413 start.push_back(0); // only always 0 when cache_slice_c_size = numids!
414 start.push_back(cache_t_idx * cache_slice_t_size);
415 count.clear();
416 count.push_back(cache_slice_c_size);
417 count.push_back(cache_slice_t_size); // Must be 1 for now!...probably...
418 ncvar.getVar(start,count,&(*cached)[0]);
419 value_cache.insert(key, cached);
420 }
421 for( size_t j = 0; j < cache_slice_t_size; j++){
422 raw_values[i+j] = cached->at((j*cache_slice_t_size) + cat_pos);
423 }
424 }
425
426
427 rvalue = 0.0;
428
429 double a , b = 0.0;
430
431 a = 1.0 - ( (t1 - init_time) / time_stride );
432 rvalue += (a * raw_values[0]);
433
434 for( size_t i = 1; i < raw_values.size() -1; ++i )
435 {
436 rvalue += raw_values[i];
437 }
438
439 if ( raw_values.size() > 1) // likewise the last data value may not be fully in the window
440 {
441 b = (stop_time - t2) / time_stride;
442 rvalue += (b * raw_values.back() );
443 }
444
445 // account for the resampling methods
446 switch(m)
447 {
448 case SUM: // we allready have the sum so do nothing
449 ;
450 break;
451
452 case MEAN:
453 {
454 // This is getting a length weighted mean
455 // the data values where allready scaled for where there was only partial use of a data value
456 // so we just need to do a final scale to account for the differnce between time_stride and duration_s
457
458 double scale_factor = (selector.get_duration_secs() > time_stride ) ? (time_stride / selector.get_duration_secs()) : (1.0 / (a + b));
459 rvalue *= scale_factor;
460 }
461 break;
462
463 default:
464 ;
465 }
466
467 try
468 {
469 return UnitsHelper::get_converted_value(native_units, rvalue, selector.get_output_units());
470 }
471 catch (const std::runtime_error& e)
472 {
473 #ifndef UDUNITS_QUIET
474 std::cerr<<"WARN: Unit conversion unsuccessful - Returning unconverted value! (\""<<e.what()<<"\")"<<std::endl;
475 #endif
476 return rvalue;
477 }
478
479 return rvalue;
480 }
