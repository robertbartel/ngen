115  size_t determine_et_formulation_index() {
116  std::shared_ptr<forcing::ForcingProvider> et_provider = nullptr;
117  if (availableData.find(NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != availableData.end()) {
118  et_provider = availableData[NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP];
119  }
120  else if (availableData.find(CSDMS_STD_NAME_POTENTIAL_ET) != availableData.end()) {
121  et_provider = availableData[CSDMS_STD_NAME_POTENTIAL_ET];
122  }
123  // The pointers in availableData and modules should be the same, based on create_multi_formulation()
124  // So, if an et_provider was found from availableData, we can compare to determine the right index.
125  if (et_provider != nullptr) {
126  for (size_t i = 0; i < modules.size(); ++i) {
127  if (et_provider == modules[i]) {
128  return i;
129  }
130  }
131  }
132  // Otherwise, check the modules directly for the standard ET output variable names.
133  for (size_t i = 0; i < modules.size(); ++i) {
134  std::vector<std::string> values = modules[i]->get_available_forcing_outputs();
135  if (std::find(values.begin(), values.end(), NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != values.end()) {
136  return i;
137  }
138  if (std::find(values.begin(), values.end(), CSDMS_STD_NAME_POTENTIAL_ET) != values.end()) {
139  return i;
140  }
141  }
142  return DEFAULT_ET_FORMULATION_INDEX;
143  }

485  template<class T>
486  std::shared_ptr<T> init_nested_module(int mod_index, std::string identifier, geojson::PropertyMap properties) {
487  std::unique_ptr<forcing::ForcingProvider> wfp = std::make_unique<forcing::WrappedForcingProvider>(this);
488  std::shared_ptr<T> mod = std::make_shared<T>(identifier, std::move(wfp), output);
489 
490  // Since this is a nested formulation, support usage of the '{{id}}' syntax for init config file paths.
491  Catchment_Formulation::config_pattern_substitution(properties, BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG,
492  "{{id}}", id);
493 
494  // Call create_formulation to perform the rest of the typical initialization steps for the formulation.
495  mod->create_formulation(properties);
496 
497  // Set this up for placing in the module_variable_maps member variable
498  std::shared_ptr<std::map<std::string, std::string>> var_aliases;
499  var_aliases = std::make_shared<std::map<std::string, std::string>>(std::map<std::string, std::string>());
500  for (const std::string &var_name : mod->get_bmi_input_variables()) {
501  std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
502  (*var_aliases)[framework_alias] = var_name;
503  // If framework_name is not in collection from which we have available data sources ...
504  if (availableData.count(framework_alias) != 1) {
505  throw std::runtime_error(
506  "Multi BMI cannot be created with module " + mod->get_model_type_name() + " with input " +
507  "variable " + framework_alias +
508  (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
509  " when there is no previously-enabled source for this input.");
510  }
511  else {
512  mod->input_forcing_providers[var_name] = availableData[framework_alias];
513  mod->input_forcing_providers[framework_alias] = availableData[framework_alias];
514  }
515  }
516 
517  // Also add the output variable aliases
518  for (const std::string &var_name : mod->get_bmi_output_variables()) {
519  std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
520  (*var_aliases)[framework_alias] = var_name;
521  if (availableData.count(framework_alias) > 0) {
522  throw std::runtime_error(
523  "Multi BMI cannot be created with module " + mod->get_model_type_name() +
524  " with output variable " + framework_alias +
525  (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
526  " because a previous module is using this output variable name/alias.");
527  }
528  availableData[framework_alias] = mod;
529  }
530  module_variable_maps[mod_index] = var_aliases;
531  return mod;
532  }
