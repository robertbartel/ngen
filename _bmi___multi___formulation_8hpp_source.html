117 size_t determine_et_formulation_index() {
118 std::shared_ptr<forcing::ForcingProvider> et_provider = nullptr;
119 if (availableData.find(NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != availableData.end()) {
120 et_provider = availableData[NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP];
121 }
122 else if (availableData.find(CSDMS_STD_NAME_POTENTIAL_ET) != availableData.end()) {
123 et_provider = availableData[CSDMS_STD_NAME_POTENTIAL_ET];
124 }
125 // The pointers in availableData and modules should be the same, based on create_multi_formulation()
126 // So, if an et_provider was found from availableData, we can compare to determine the right index.
127 if (et_provider != nullptr) {
128 for (size_t i = 0; i < modules.size(); ++i) {
129 if (et_provider == modules[i]) {
130 return i;
131 }
132 }
133 }
134 // Otherwise, check the modules directly for the standard ET output variable names.
135 for (size_t i = 0; i < modules.size(); ++i) {
136 std::vector<std::string> values = modules[i]->get_available_forcing_outputs();
137 if (std::find(values.begin(), values.end(), NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != values.end()) {
138 return i;
139 }
140 if (std::find(values.begin(), values.end(), CSDMS_STD_NAME_POTENTIAL_ET) != values.end()) {
141 return i;
142 }
143 }
144 return DEFAULT_ET_FORMULATION_INDEX;

583 template<class T>
584 std::shared_ptr<T> init_nested_module(int mod_index, std::string identifier, geojson::PropertyMap properties) {
585 std::unique_ptr<forcing::ForcingProvider> wfp = std::make_unique<forcing::WrappedForcingProvider>(this);
586 std::shared_ptr<T> mod = std::make_shared<T>(identifier, std::move(wfp), output);
587
588 // Since this is a nested formulation, support usage of the '{{id}}' syntax for init config file paths.
589 Catchment_Formulation::config_pattern_substitution(properties, BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG,
590 "{{id}}", id);
591
592 // Call create_formulation to perform the rest of the typical initialization steps for the formulation.
593 mod->create_formulation(properties);
594
595 // Set this up for placing in the module_variable_maps member variable
596 std::shared_ptr<std::map<std::string, std::string>> var_aliases;
597 var_aliases = std::make_shared<std::map<std::string, std::string>>(std::map<std::string, std::string>());
598 for (const std::string &var_name : mod->get_bmi_input_variables()) {
599 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
600 (*var_aliases)[framework_alias] = var_name;
601 // If framework_name is not in collection from which we have available data sources ...
602 if (availableData.count(framework_alias) != 1) {
603 throw std::runtime_error(
604 "Multi BMI cannot be created with module " + mod->get_model_type_name() + " with input " +
605 "variable " + framework_alias +
606 (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
607 " when there is no previously-enabled source for this input.");
608 }
609 else {
610 mod->input_forcing_providers[var_name] = availableData[framework_alias];
611 mod->input_forcing_providers[framework_alias] = availableData[framework_alias];
612 }
613 }
614
615 // Also add the output variable aliases
616 for (const std::string &var_name : mod->get_bmi_output_variables()) {
617 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
618 (*var_aliases)[framework_alias] = var_name;
619 if (availableData.count(framework_alias) > 0) {
620 throw std::runtime_error(
621 "Multi BMI cannot be created with module " + mod->get_model_type_name() +
622 " with output variable " + framework_alias +
623 (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
624 " because a previous module is using this output variable name/alias.");
625 }
626 availableData[framework_alias] = mod;
627 }
628 module_variable_maps[mod_index] = var_aliases;
629 return mod;
630 }
