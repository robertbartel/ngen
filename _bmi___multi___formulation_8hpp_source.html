79  size_t determine_et_formulation_index() {
80  std::shared_ptr<forcing::ForcingProvider> et_provider = nullptr;
81  if (availableData.find(NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != availableData.end()) {
82  et_provider = availableData[NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP];
83  }
84  else if (availableData.find(CSDMS_STD_NAME_POTENTIAL_ET) != availableData.end()) {
85  et_provider = availableData[CSDMS_STD_NAME_POTENTIAL_ET];
86  }
87  // The pointers in availableData and modules should be the same, based on create_multi_formulation()
88  // So, if an et_provider was found from availableData, we can compare to determine the right index.
89  if (et_provider != nullptr) {
90  for (size_t i = 0; i < modules.size(); ++i) {
91  if (et_provider == modules[i]) {
92  return i;
93  }
94  }
95  }
96  // Otherwise, check the modules directly for the standard ET output variable names.
97  for (size_t i = 0; i < modules.size(); ++i) {
98  std::vector<std::string> values = modules[i]->get_available_forcing_outputs();
99  if (std::find(values.begin(), values.end(), NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != values.end()) {
100  return i;
101  }
102  if (std::find(values.begin(), values.end(), CSDMS_STD_NAME_POTENTIAL_ET) != values.end()) {
103  return i;
104  }
105  }
106  return DEFAULT_ET_FORMULATION_INDEX;
107  }
