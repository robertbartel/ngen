120 size_t determine_et_formulation_index() {
121 std::shared_ptr<forcing::ForcingProvider> et_provider = nullptr;
122 if (availableData.find(NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != availableData.end()) {
123 et_provider = availableData[NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP];
124 }
125 else if (availableData.find(CSDMS_STD_NAME_POTENTIAL_ET) != availableData.end()) {
126 et_provider = availableData[CSDMS_STD_NAME_POTENTIAL_ET];
127 }
128 // The pointers in availableData and modules should be the same, based on create_multi_formulation()
129 // So, if an et_provider was found from availableData, we can compare to determine the right index.
130 if (et_provider != nullptr) {
131 for (size_t i = 0; i < modules.size(); ++i) {
132 if (et_provider == modules[i]) {
133 return i;
134 }
135 }
136 }
137 // Otherwise, check the modules directly for the standard ET output variable names.
138 for (size_t i = 0; i < modules.size(); ++i) {
139 std::vector<std::string> values = modules[i]->get_available_forcing_outputs();
140 if (std::find(values.begin(), values.end(), NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != values.end()) {
141 return i;
142 }
143 if (std::find(values.begin(), values.end(), CSDMS_STD_NAME_POTENTIAL_ET) != values.end()) {
144 return i;
145 }
146 }
147 return DEFAULT_ET_FORMULATION_INDEX;

580 inline void init_deferred_associations() {
581 for (int d = 0; d < deferredProviders.size(); ++d) {
582 std::shared_ptr<forcing::OptionalWrappedProvider> &deferredProvider = deferredProviders[d];
583 // Skip doing anything for any deferred provider that already has its backing provider set
584 if (deferredProvider->isWrappedProviderSet())
585 continue;
586
587 // TODO: improve this later; since defaults can be used, it is technically possible to grab something
588 // valid when something more appropriate would later be available
589 // Iterate through available data providers and set association once a sufficient one is found
590 std::map<std::string, std::shared_ptr<forcing::ForcingProvider>>::iterator avail_it;
591 for (avail_it = availableData.begin(); avail_it != availableData.end(); avail_it++) {
592 // If this satisfies everything this deferred provider needs to provide, and thus can be/was set ...
593 if (deferredProvider->setWrappedProvider(avail_it->second.get())) {
594 break;
595 }
596 }
597
598 // If none of the available data providers could be used for this deferred one, throw exception
599 if (!deferredProvider->isWrappedProviderSet()) {
600 // TODO: this probably needs to be some kind of custom configuration exception
601 std::string msg = "Multi BMI formulation cannot be created from config: cannot find available data "
602 "provider to satisfy set of deferred provisions for nested module at index "
603 + std::to_string(deferredProviderModuleIndices[d]) + ": {";
604 // There must always be at least 1; get manually to help with formatting
605 msg += deferredProvider->get_available_forcing_outputs()[0];
606 // And here make sure to start at 1 instead of 0
607 for (int i = 1; i < deferredProvider->get_available_forcing_outputs().size(); ++i)
608 msg += ", " + deferredProvider->get_available_forcing_outputs()[i];
609 msg += "}";
610 throw realization::ConfigurationException(msg);
611 }
612 }

630 template<class T>
631 std::shared_ptr<T> init_nested_module(int mod_index, std::string identifier, geojson::PropertyMap properties) {
632 std::unique_ptr<forcing::ForcingProvider> wfp = std::make_unique<forcing::WrappedForcingProvider>(this);
633 std::shared_ptr<T> mod = std::make_shared<T>(identifier, std::move(wfp), output);
634
635 // Since this is a nested formulation, support usage of the '{{id}}' syntax for init config file paths.
636 Catchment_Formulation::config_pattern_substitution(properties, BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG,
637 "{{id}}", id);
638
639 // Call create_formulation to perform the rest of the typical initialization steps for the formulation.
640 mod->create_formulation(properties);
641
642 // Set this up for placing in the module_variable_maps member variable
643 std::shared_ptr<std::map<std::string, std::string>> var_aliases;
644 var_aliases = std::make_shared<std::map<std::string, std::string>>(std::map<std::string, std::string>());
645 for (const std::string &var_name : mod->get_bmi_input_variables()) {
646 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
647 (*var_aliases)[framework_alias] = var_name;
648 // If framework_name is not yet in collection from which we have available data sources ...
649 if (availableData.count(framework_alias) != 1) {
650 setup_nested_deferred_provider(var_name, framework_alias, mod, mod_index);
651 }
652 else {
653 mod->input_forcing_providers[var_name] = availableData[framework_alias];
654 mod->input_forcing_providers[framework_alias] = availableData[framework_alias];
655 }
656 }
657
658 // Also add the output variable aliases
659 for (const std::string &var_name : mod->get_bmi_output_variables()) {
660 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
661 (*var_aliases)[framework_alias] = var_name;
662 if (availableData.count(framework_alias) > 0) {
663 throw std::runtime_error(
664 "Multi BMI cannot be created with module " + mod->get_model_type_name() +
665 " with output variable " + framework_alias +
666 (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
667 " because a previous module is using this output variable name/alias.");
668 }
669 availableData[framework_alias] = mod;
670 }
671 module_variable_maps[mod_index] = var_aliases;
672 return mod;
673 }

702 template<class T>
703 void setup_nested_deferred_provider(const std::string &bmi_input_var_name,
704 const std::string &framework_output_name,
705 std::shared_ptr<T> mod,
706 int mod_index) {
707 // TODO: probably don't actually need bmi_input_var_name, and just can deal with framework_output_name
708 // Create deferred, optional provider for providing this
709 // Only include BMI variable name, as that's what'll be visible when associating to backing provider
710 // It's "deferred" in that we'll set the backing later.
711 // It's "optional" in that it waits to use backing provider, using the default some number of times
712 std::shared_ptr<forcing::OptionalWrappedProvider> provider;
713 // TODO: make sure only alias is needed
714 auto defs_it = default_output_values.find(framework_output_name);
715 if (defs_it != default_output_values.end()) {
716 // TODO: consider also reading wait count from config
717 provider = std::make_shared<forcing::OptionalWrappedProvider>(framework_output_name, defs_it->second, 1);
718 }
719 else {
720 provider = std::make_shared<forcing::OptionalWrappedProvider>(framework_output_name);
721 }
722
723 // Add deferred to collection and module index to collection
724 deferredProviders.push_back(provider);
725 deferredProviderModuleIndices.push_back(mod_index);
726 // Assign as provider within module
727 // TODO: per TODO at top, probably can replace bmi_input_var_name here with framework_output_name
728 mod->input_forcing_providers[bmi_input_var_name] = provider;
729 }
