120 size_t determine_et_formulation_index() {
121 std::shared_ptr<data_access::GenericDataProvider> et_provider = nullptr;
122 if (availableData.find(NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != availableData.end()) {
123 et_provider = availableData[NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP];
124 }
125 else if (availableData.find(CSDMS_STD_NAME_POTENTIAL_ET) != availableData.end()) {
126 et_provider = availableData[CSDMS_STD_NAME_POTENTIAL_ET];
127 }
128 // The pointers in availableData and modules should be the same, based on create_multi_formulation()
129 // So, if an et_provider was found from availableData, we can compare to determine the right index.
130 if (et_provider != nullptr) {
131 for (size_t i = 0; i < modules.size(); ++i) {
132 if (et_provider == modules[i]) {
133 return i;
134 }
135 }
136 }
137 // Otherwise, check the modules directly for the standard ET output variable names.
138 for (size_t i = 0; i < modules.size(); ++i) {
139 std::vector<std::string> values = modules[i]->get_avaliable_variable_names();
140 if (std::find(values.begin(), values.end(), NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != values.end()) {
141 return i;
142 }
143 if (std::find(values.begin(), values.end(), CSDMS_STD_NAME_POTENTIAL_ET) != values.end()) {
144 return i;
145 }
146 }
147 return DEFAULT_ET_FORMULATION_INDEX;
148 }

654 inline void init_deferred_associations() {
655 for (int d = 0; d < deferredProviders.size(); ++d) {
656 std::shared_ptr<data_access::OptionalWrappedProvider> &deferredProvider = deferredProviders[d];
657 // Skip doing anything for any deferred provider that already has its backing provider set
658 if (deferredProvider->isWrappedProviderSet())
659 continue;
660
661 // TODO: improve this later; since defaults can be used, it is technically possible to grab something
662 // valid when something more appropriate would later be available
663 // Iterate through available data providers and set association once a sufficient one is found
664 std::map<std::string, std::shared_ptr<data_access::GenericDataProvider>>::iterator avail_it;
665 for (avail_it = availableData.begin(); avail_it != availableData.end(); avail_it++) {
666 // If this satisfies everything this deferred provider needs to provide, and thus can be/was set ...
667 if (deferredProvider->setWrappedProvider(avail_it->second.get())) {
668 break;
669 }
670 }
671
672 // If none of the available data providers could be used for this deferred one, throw exception
673 if (!deferredProvider->isWrappedProviderSet()) {
674 // TODO: this probably needs to be some kind of custom configuration exception
675 std::string msg = "Multi BMI formulation cannot be created from config: cannot find available data "
676 "provider to satisfy set of deferred provisions for nested module at index "
677 + std::to_string(deferredProviderModuleIndices[d]) + ": {";
678 // There must always be at least 1; get manually to help with formatting
679 msg += deferredProvider->get_avaliable_variable_names()[0];
680 // And here make sure to start at 1 instead of 0
681 for (int i = 1; i < deferredProvider->get_avaliable_variable_names().size(); ++i)
682 msg += ", " + deferredProvider->get_avaliable_variable_names()[i];
683 msg += "}";
684 throw realization::ConfigurationException(msg);
685 }
686 }

704 template<class T>
705 std::shared_ptr<T> init_nested_module(int mod_index, std::string identifier, geojson::PropertyMap properties) {
706 std::shared_ptr<data_access::GenericDataProvider> wfp = std::make_shared<data_access::WrappedForcingProvider>(this);
707 std::shared_ptr<T> mod = std::make_shared<T>(identifier, wfp, output);
708
709 // Since this is a nested formulation, support usage of the '{{id}}' syntax for init config file paths.
710 Catchment_Formulation::config_pattern_substitution(properties, BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG,
711 "{{id}}", id);
712
713 // Call create_formulation to perform the rest of the typical initialization steps for the formulation.
714 mod->create_formulation(properties);
715
716 // Set this up for placing in the module_variable_maps member variable
717 std::shared_ptr<std::map<std::string, std::string>> var_aliases;
718 var_aliases = std::make_shared<std::map<std::string, std::string>>(std::map<std::string, std::string>());
719 for (const std::string &var_name : mod->get_bmi_input_variables()) {
720 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
721 (*var_aliases)[framework_alias] = var_name;
722 // If framework_name is not yet in collection from which we have available data sources ...
723 if (availableData.count(framework_alias) != 1) {
724 setup_nested_deferred_provider(var_name, framework_alias, mod, mod_index);
725 }
726 else {
727 mod->input_forcing_providers[var_name] = availableData[framework_alias];
728 mod->input_forcing_providers[framework_alias] = availableData[framework_alias];
729 }
730 }
731
732 // Also add the output variable aliases
733 for (const std::string &var_name : mod->get_bmi_output_variables()) {
734 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
735 (*var_aliases)[framework_alias] = var_name;
736 if (availableData.count(framework_alias) > 0) {
737 throw std::runtime_error(
738 "Multi BMI cannot be created with module " + mod->get_model_type_name() +
739 " with output variable " + framework_alias +
740 (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
741 " because a previous module is using this output variable name/alias.");
742 }
743 availableData[framework_alias] = mod;
744 }
745 module_variable_maps[mod_index] = var_aliases;
746 return mod;
747 }

776 template<class T>
777 void setup_nested_deferred_provider(const std::string &bmi_input_var_name,
778 const std::string &framework_output_name,
779 std::shared_ptr<T> mod,
780 int mod_index) {
781 // TODO: probably don't actually need bmi_input_var_name, and just can deal with framework_output_name
782 // Create deferred, optional provider for providing this
783 // Only include BMI variable name, as that's what'll be visible when associating to backing provider
784 // It's "deferred" in that we'll set the backing later.
785 // It's "optional" in that it waits to use backing provider, using the default some number of times
786 std::shared_ptr<data_access::OptionalWrappedProvider> provider;
787 // TODO: make sure only alias is needed
788 auto defs_it = default_output_values.find(framework_output_name);
789 if (defs_it != default_output_values.end()) {
790 // TODO: consider also reading wait count from config
791 provider = std::make_shared<data_access::OptionalWrappedProvider>(framework_output_name, defs_it->second, 1);
792 }
793 else {
794 provider = std::make_shared<data_access::OptionalWrappedProvider>(framework_output_name);
795 }
796
797 // Add deferred to collection and module index to collection
798 deferredProviders.push_back(provider);
799 deferredProviderModuleIndices.push_back(mod_index);
800 // Assign as provider within module
801 // TODO: per TODO at top, probably can replace bmi_input_var_name here with framework_output_name
802 mod->input_forcing_providers[bmi_input_var_name] = provider;
803 }
