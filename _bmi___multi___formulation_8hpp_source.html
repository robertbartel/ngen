121 size_t determine_et_formulation_index() {
122 std::shared_ptr<data_access::GenericDataProvider> et_provider = nullptr;
123 if (availableData.find(NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != availableData.end()) {
124 et_provider = availableData[NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP];
125 }
126 else if (availableData.find(CSDMS_STD_NAME_POTENTIAL_ET) != availableData.end()) {
127 et_provider = availableData[CSDMS_STD_NAME_POTENTIAL_ET];
128 }
129 // The pointers in availableData and modules should be the same, based on create_multi_formulation()
130 // So, if an et_provider was found from availableData, we can compare to determine the right index.
131 if (et_provider != nullptr) {
132 for (size_t i = 0; i < modules.size(); ++i) {
133 if (et_provider == modules[i]) {
134 return i;
135 }
136 }
137 }
138 // Otherwise, check the modules directly for the standard ET output variable names.
139 for (size_t i = 0; i < modules.size(); ++i) {
140 std::vector<std::string> values = modules[i]->get_avaliable_variable_names();
141 if (std::find(values.begin(), values.end(), NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != values.end()) {
142 return i;
143 }
144 if (std::find(values.begin(), values.end(), CSDMS_STD_NAME_POTENTIAL_ET) != values.end()) {
145 return i;
146 }
147 }
148 return DEFAULT_ET_FORMULATION_INDEX;

669 inline void init_deferred_associations() {
670 for (int d = 0; d < deferredProviders.size(); ++d) {
671 std::shared_ptr<data_access::OptionalWrappedDataProvider> &deferredProvider = deferredProviders[d];
672 // Skip doing anything for any deferred provider that already has its backing provider set
673 if (deferredProvider->isWrappedProviderSet())
674 continue;
675
676 // TODO: improve this later; since defaults can be used, it is technically possible to grab something
677 // valid when something more appropriate would later be available
678 // Iterate through available data providers and set association once a sufficient one is found
679 std::map<std::string, std::shared_ptr<data_access::GenericDataProvider>>::iterator avail_it;
680 for (avail_it = availableData.begin(); avail_it != availableData.end(); avail_it++) {
681 // If this satisfies everything this deferred provider needs to provide, and thus can be/was set ...
682 if (deferredProvider->setWrappedProvider(avail_it->second.get())) {
683 break;
684 }
685 }
686
687 // If none of the available data providers could be used for this deferred one, throw exception
688 if (!deferredProvider->isWrappedProviderSet()) {
689 // TODO: this probably needs to be some kind of custom configuration exception
690 std::string msg = "Multi BMI formulation cannot be created from config: cannot find available data "
691 "provider to satisfy set of deferred provisions for nested module at index "
692 + std::to_string(deferredProviderModuleIndices[d]) + ": {";
693 // There must always be at least 1; get manually to help with formatting
694 msg += deferredProvider->get_avaliable_variable_names()[0];
695 // And here make sure to start at 1 instead of 0
696 for (int i = 1; i < deferredProvider->get_avaliable_variable_names().size(); ++i)
697 msg += ", " + deferredProvider->get_avaliable_variable_names()[i];
698 msg += "}";
699 throw realization::ConfigurationException(msg);
700 }
701 }

719 template<class T>
720 std::shared_ptr<T> init_nested_module(int mod_index, std::string identifier, geojson::PropertyMap properties) {
721 std::shared_ptr<data_access::GenericDataProvider> wfp = std::make_shared<data_access::WrappedDataProvider>(this);
722 std::shared_ptr<T> mod = std::make_shared<T>(identifier, wfp, output);
723
724 // Since this is a nested formulation, support usage of the '{{id}}' syntax for init config file paths.
725 Catchment_Formulation::config_pattern_substitution(properties, BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG,
726 "{{id}}", id);
727
728 // Call create_formulation to perform the rest of the typical initialization steps for the formulation.
729 mod->create_formulation(properties);
730
731 // Set this up for placing in the module_variable_maps member variable
732 std::shared_ptr<std::map<std::string, std::string>> var_aliases;
733 var_aliases = std::make_shared<std::map<std::string, std::string>>(std::map<std::string, std::string>());
734 for (const std::string &var_name : mod->get_bmi_input_variables()) {
735 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
736 (*var_aliases)[framework_alias] = var_name;
737 // If framework_name is not yet in collection from which we have available data sources ...
738 if (availableData.count(framework_alias) != 1) {
739 setup_nested_deferred_provider(var_name, framework_alias, mod, mod_index);
740 }
741 else {
742 mod->input_forcing_providers[var_name] = availableData[framework_alias];
743 mod->input_forcing_providers[framework_alias] = availableData[framework_alias];
744 }
745 }
746
747 // Also add the output variable aliases
748 for (const std::string &var_name : mod->get_bmi_output_variables()) {
749 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
750 (*var_aliases)[framework_alias] = var_name;
751 if (availableData.count(framework_alias) > 0) {
752 throw std::runtime_error(
753 "Multi BMI cannot be created with module " + mod->get_model_type_name() +
754 " with output variable " + framework_alias +
755 (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
756 " because a previous module is using this output variable name/alias.");
757 }
758 availableData[framework_alias] = mod;
759 }
760 module_variable_maps[mod_index] = var_aliases;
761 return mod;
762 }

792 void setup_nested_deferred_provider(const std::string &bmi_input_var_name,
793 const std::string &framework_output_name,
794 std::shared_ptr<T> mod,
795 int mod_index) {
796 // TODO: probably don't actually need bmi_input_var_name, and just can deal with framework_output_name
797 // Create deferred, optional provider for providing this
798 // Only include BMI variable name, as that's what'll be visible when associating to backing provider
799 // It's "deferred" in that we'll set the backing later.
800 // It's "optional" in that it waits to use backing provider, using the default some number of times
801 std::shared_ptr<data_access::OptionalWrappedDataProvider> provider;
802 // TODO: make sure only alias is needed
803 auto defs_it = default_output_values.find(framework_output_name);
804 if (defs_it != default_output_values.end()) {
805 // TODO: consider also reading wait count from config
806 provider = std::make_shared<data_access::OptionalWrappedDataProvider>(framework_output_name, defs_it->second, 1);
807 }
808 else {
809 provider = std::make_shared<data_access::OptionalWrappedDataProvider>(framework_output_name);
810 }
811
812 // Add deferred to collection and module index to collection
813 deferredProviders.push_back(provider);
814 deferredProviderModuleIndices.push_back(mod_index);
815 // Assign as provider within module
816 // TODO: per TODO at top, probably can replace bmi_input_var_name here with framework_output_name
817 mod->input_forcing_providers[bmi_input_var_name] = provider;
818 }
