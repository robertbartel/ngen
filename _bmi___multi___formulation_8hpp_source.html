117  size_t determine_et_formulation_index() {
118  std::shared_ptr<forcing::ForcingProvider> et_provider = nullptr;
119  if (availableData.find(NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != availableData.end()) {
120  et_provider = availableData[NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP];
121  }
122  else if (availableData.find(CSDMS_STD_NAME_POTENTIAL_ET) != availableData.end()) {
123  et_provider = availableData[CSDMS_STD_NAME_POTENTIAL_ET];
124  }
125  // The pointers in availableData and modules should be the same, based on create_multi_formulation()
126  // So, if an et_provider was found from availableData, we can compare to determine the right index.
127  if (et_provider != nullptr) {
128  for (size_t i = 0; i < modules.size(); ++i) {
129  if (et_provider == modules[i]) {
130  return i;
131  }
132  }
133  }
134  // Otherwise, check the modules directly for the standard ET output variable names.
135  for (size_t i = 0; i < modules.size(); ++i) {
136  std::vector<std::string> values = modules[i]->get_available_forcing_outputs();
137  if (std::find(values.begin(), values.end(), NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != values.end()) {
138  return i;
139  }
140  if (std::find(values.begin(), values.end(), CSDMS_STD_NAME_POTENTIAL_ET) != values.end()) {
141  return i;
142  }
143  }
144  return DEFAULT_ET_FORMULATION_INDEX;
145  }

524  template<class T>
525  std::shared_ptr<T> init_nested_module(int mod_index, std::string identifier, geojson::PropertyMap properties) {
526  std::unique_ptr<forcing::ForcingProvider> wfp = std::make_unique<forcing::WrappedForcingProvider>(this);
527  std::shared_ptr<T> mod = std::make_shared<T>(identifier, std::move(wfp), output);
528 
529  // Since this is a nested formulation, support usage of the '{{id}}' syntax for init config file paths.
530  Catchment_Formulation::config_pattern_substitution(properties, BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG,
531  "{{id}}", id);
532 
533  // Call create_formulation to perform the rest of the typical initialization steps for the formulation.
534  mod->create_formulation(properties);
535 
536  // Set this up for placing in the module_variable_maps member variable
537  std::shared_ptr<std::map<std::string, std::string>> var_aliases;
538  var_aliases = std::make_shared<std::map<std::string, std::string>>(std::map<std::string, std::string>());
539  for (const std::string &var_name : mod->get_bmi_input_variables()) {
540  std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
541  (*var_aliases)[framework_alias] = var_name;
542  // If framework_name is not in collection from which we have available data sources ...
543  if (availableData.count(framework_alias) != 1) {
544  throw std::runtime_error(
545  "Multi BMI cannot be created with module " + mod->get_model_type_name() + " with input " +
546  "variable " + framework_alias +
547  (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
548  " when there is no previously-enabled source for this input.");
549  }
550  else {
551  mod->input_forcing_providers[var_name] = availableData[framework_alias];
552  mod->input_forcing_providers[framework_alias] = availableData[framework_alias];
553  }
554  }
555 
556  // Also add the output variable aliases
557  for (const std::string &var_name : mod->get_bmi_output_variables()) {
558  std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
559  (*var_aliases)[framework_alias] = var_name;
560  if (availableData.count(framework_alias) > 0) {
561  throw std::runtime_error(
562  "Multi BMI cannot be created with module " + mod->get_model_type_name() +
563  " with output variable " + framework_alias +
564  (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
565  " because a previous module is using this output variable name/alias.");
566  }
567  availableData[framework_alias] = mod;
568  }
569  module_variable_maps[mod_index] = var_aliases;
570  return mod;
571  }
