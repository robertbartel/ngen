71 size_t get_ts_index_for_time(const time_t &epoch_time) override {
72 if (epoch_time < start_date_time_epoch) {
73 throw std::out_of_range("Forcing had bad pre-start time for index query: " + std::to_string(epoch_time));
74 }
75 size_t i = 0;
76 // 1 hour
77 time_t seconds_in_time_step = 3600;
78 time_t time = start_date_time_epoch;
79 while (epoch_time >= time + seconds_in_time_step && time < end_date_time_epoch) {
80 i++;
81 time += seconds_in_time_step;
82 }
83 // The end_date_time_epoch is the epoch value of the BEGINNING of the last time step, not its end.
84 // I.e., to make sure we cover it, we have to go another time step beyond.
85 if (time >= end_date_time_epoch + 3600) {
86 throw std::out_of_range("Forcing had bad beyond-end time for index query: " + std::to_string(epoch_time));
87 }
88 else {
89 return i;
90 }

105 double get_value(const std::string &output_name, const time_t &init_time, const long &duration_s,
106 const std::string &output_units) override
107 {
108 size_t current_index;
109 long time_remaining = duration_s;
110 try {
111 current_index = get_ts_index_for_time(init_time);
112 }
113 catch (const std::out_of_range &e) {
114 throw std::out_of_range("Forcing had bad init_time " + std::to_string(init_time) + " for value request");
115 }
116
117 std::vector<double> involved_time_step_values;
118 std::vector<long> involved_time_step_seconds;
119 long ts_involved_s;
120
121 time_t first_time_step_start_epoch = start_date_time_epoch + (current_index * 3600);
122 // Handle the first time step differently, since we need to do more to figure out how many seconds came from it
123 // Total time step size minus the offset of the beginning, before the init time
124 ts_involved_s = 3600 - (init_time - first_time_step_start_epoch);
125
126 involved_time_step_seconds.push_back(ts_involved_s);
127 involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
128 time_remaining -= ts_involved_s;
129 current_index++;
130
131 while (time_remaining > 0) {
132 if(current_index >= time_epoch_vector.size())
133 return involved_time_step_values[involved_time_step_values.size()-1]; //TODO: Is this the right answer? Is returning any value off the end of the range valid?
134 ts_involved_s = time_remaining > 3600 ? 3600 : time_remaining;
135 involved_time_step_seconds.push_back(ts_involved_s);
136 involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
137 time_remaining -= ts_involved_s;
138 current_index++;
139
140 }
141 double value = 0;
142 for (size_t i = 0; i < involved_time_step_values.size(); ++i) {
143 if (is_param_sum_over_time_step(output_name))
144 value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / 3600.0);
145 else
146 value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / (double)duration_s);
147 }
148
149 // Convert units
150 try {
151 return UnitsHelper::get_converted_value(available_forcings_units[output_name], value, output_units);
152 }
153 catch (const std::runtime_error& e){
154 #ifndef UDUNITS_QUIET
155 std::cerr<<"Unit conversion error: "<<std::endl<<e.what()<<std::endl<<"Returning unconverted value!"<<std::endl;
156 #endif
157 return value;
158 }

237 inline void check_forcing_vector_index_bounds()
238 {
239 //Check if forcing index is less than zero and if so, set to zero.
240 if (forcing_vector_index < 0)
241 {
242 forcing_vector_index = 0;
244 cout << "WARNING: Forcing vector index is less than zero. Therefore, setting index to zero." << endl;
245 }
246
247 //Check if forcing index is greater than or equal to the size of the size of the time vector and if so, set to zero.
248 else if (forcing_vector_index >= time_epoch_vector.size())
249 {
250 forcing_vector_index = time_epoch_vector.size() - 1;
252 cout << "WARNING: Reached beyond the size of the forcing vector. Therefore, setting index to last value of the vector." << endl;
253 }
254
255 return;

289 void read_csv(string file_name)
290 {
291 int time_col_index = 0;
292 //std::map<std::string, int> col_indices;
293 std::vector<std::vector<double>*> local_valvec_index = {};
294
295 //Call CSVReader constuctor
296 CSVReader reader(file_name);
297
298 //Get the data from CSV File
299 std::vector<std::vector<std::string> > data_list = reader.getData();
300
301 // Process the header (first) row..
302 int col_num = 0;
303 for (const auto& col_head : data_list[0]){
304 //std::cerr << s << std::endl;
305 if(col_head == "Time" || col_head == "time"){
306 time_col_index = col_num;
307 local_valvec_index.push_back(nullptr); // make sure the column indices line up!
308 } else {
309 std::string var_name = col_head;
310 std::string units = "";
311
312 //TODO: parse units in parens and/or square brackets?
313
314 if(well_known_fields.count(var_name) > 0){
315 units = units.empty() ? std::get<1>(well_known_fields[var_name]) : units;
316 available_forcings.push_back(var_name); // Allow lookup by non-canonical name
317 available_forcings_units[var_name] = units; // Allow lookup of units by non-canonical name
318 var_name = std::get<0>(well_known_fields[var_name]); // Use the CSDMS name from here on
319 }
320
321 forcing_vectors[var_name] = {};
322 local_valvec_index.push_back(&(forcing_vectors[var_name]));
323 available_forcings.push_back(var_name);
324 available_forcings_units[var_name] = units;
325 }
326 col_num++;
327 }
328
329 time_t current_row_date_time_epoch;
330 //Iterate through CSV starting on the second row
331 int i = 1;
332 for (i = 1; i < data_list.size(); i++)
333 {
334 //Row vector
335 std::vector<std::string>& vec = data_list[i];
336
337 struct tm current_row_date_time_utc = tm();
338 string time_str = vec[time_col_index];
339 //TODO: Support more time string formats? This is basically ISO8601 but not complete, support TZ?
340 strptime(time_str.c_str(), "%Y-%m-%d %H:%M:%S", &current_row_date_time_utc);
341
342 //Convert current row date-time UTC to epoch time
343 current_row_date_time_epoch = timegm(&current_row_date_time_utc);
344
345 //TODO: I am not sure this is a concern of this object. If forcing is retrieved that doesn't cover the
346 //needed time period, isn't that the requester's concern? (Methods exist to check this...)
347 //Ensure that forcing data covers the entire model period. Otherwise, throw an error.
348 if (i == 1 && start_date_time_epoch < current_row_date_time_epoch)
349 {
350 struct tm start_date_tm;
351 gmtime_r(&start_date_time_epoch, &start_date_tm);
352
353 char tm_buff[128];
354 strftime(tm_buff, 128, "%Y-%m-%d %H:%M:%S", &start_date_tm);
355 throw std::runtime_error("Error: Forcing data " + file_name + " begins after the model start time:" + std::string(tm_buff) + " < " + time_str);
356 }
357
358
359 if (start_date_time_epoch <= current_row_date_time_epoch && current_row_date_time_epoch <= end_date_time_epoch)
360 {
361 time_epoch_vector.push_back(current_row_date_time_epoch);
362
363 int c = -1;
364 for (auto& s : vec){
365 c++;
366 if(c == time_col_index)
367 continue;
368 boost::algorithm::trim(s);
369 local_valvec_index[c]->push_back(boost::lexical_cast<double>(s)); // This is supposed to update the vector in the map...
370 }
371
372 }
373
374 }
375 if (i <= 1 || current_row_date_time_epoch < end_date_time_epoch)
376 {
378 cout << "WARNING: Forcing data ends before the model end time." << endl;
379 //throw std::runtime_error("Error: Forcing data ends before the model end time.");
380 }
381 }
