221  const std::string get_analogous_cxx_type(const std::string &py_type_name, const size_t item_size) override {
222  /*
223  * Note that an implementation using a "switch" statement would be problematic. It could be done by
224  * rewriting to separate the integer and non-integer type, then having cases based on size. However,
225  * this might cause trouble on certain systems, since (depending on the particular sizes of types) that
226  * could produce duplicate "case" values.
227  */
228  if (py_type_name == "int" && item_size == sizeof(short)) {
229  return "short";
230  } else if (py_type_name == "int" && item_size == sizeof(int)) {
231  return "int";
232  } else if (py_type_name == "int" && item_size == sizeof(long)) {
233  return "long";
234  } else if (py_type_name == "int" && item_size == sizeof(long long)) {
235  return "long long";
236  } else if (py_type_name == "float" && item_size == sizeof(float)) {
237  return "float";
238  } else if ((py_type_name == "float" || py_type_name == "float64" || py_type_name == "np.float64" ||
239  py_type_name == "numpy.float64") && item_size == sizeof(double)) {
240  return "double";
241  } else if (py_type_name == "float" && item_size == sizeof(long double)) {
242  return "long double";
243  } else {
244  throw runtime_error(
245  "(Bmi_Py_Adapter) Failed determining analogous C++ type for Python model '" + py_type_name +
246  "' type with size " + std::to_string(item_size) + " bytes.");
247  }
248  }

665  inline void separate_package_and_simple_name() {
666  if (!model_initialized) {
667  vector<string> split_name;
668  string delimiter = ".";
669  string name_string = bmi_type_py_full_name;
670 
671  size_t pos = 0;
672  string token;
673  while ((pos = name_string.find(delimiter)) != string::npos) {
674  token = name_string.substr(0, pos);
675  split_name.emplace_back(token);
676  name_string.erase(0, pos + delimiter.length());
677  }
678  if (split_name.empty()) {
679  throw std::runtime_error("Cannot interpret BMI Python model type '" + bmi_type_py_full_name
680  + "'; expected format is <python_module>.<python_class>");
681  }
682  // What's left should be the class name
683  bmi_type_py_class_name = make_shared<string>(name_string);
684  //split_name.pop_back();
685  // And then the split name should contain the module
686  // TODO: going to need to look at this again in the future; right now, assuming the format
687  // <python_module>.<python_class> works fine as long as a model class is always in a top-level
688  // module, but the current logic is going to interpret any complex parent module name as a single
689  // top-level namespace package; e.g., ngen.namespacepackage.model works if ngen.namespacepackage is
690  // a namespace package, but there would be problems with something like ngenpkg.innermodule1.model.
691  bmi_type_py_module_name = make_shared<string>(boost::algorithm::join(split_name, delimiter));
692  }
