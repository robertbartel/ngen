643  std::shared_ptr<std::vector<std::string>> inner_get_variable_names(bool is_input_variables) {
644  // Writing this def once here at beginning: fewer lines, but this may or may not be used.
645  std::string varType = (is_input_variables) ? "input" : "output";
646 
647  // Obtain this via inner functions, which should use the model directly and not other member functions
648  int variableCount;
649  try {
650  variableCount = (is_input_variables) ? inner_get_input_item_count() : inner_get_output_item_count();
651  }
652  catch (const std::exception &e) {
653  throw std::runtime_error(model_name + " failed to count of " + varType + " variable names array.");
654  }
655 
656  // With variable count now obtained, create the vector
657  std::shared_ptr<std::vector<std::string>> var_names = std::make_shared<std::vector<std::string>>(
658  std::vector<std::string>(variableCount));
659 
660  // Must get the names from the model as an array of C strings
661  // The array can be on the stack ...
662  char* names_array[variableCount];
663  // ... but allocate the space for the individual C strings (i.e., the char * elements)
664  for (int i = 0; i < variableCount; i++) {
665  names_array[i] = static_cast<char *>(malloc(sizeof(char) * BMI_MAX_VAR_NAME));
666  }
667 
668  // With the necessary char** in hand, get the names from the model
669  int names_result;
670  if (is_input_variables) {
671  names_result = get_input_var_names(bmi_model.get(), names_array);
672  }
673  else {
674  names_result = get_output_var_names(bmi_model.get(), names_array);
675  }
676  if (names_result != BMI_SUCCESS) {
677  throw std::runtime_error(model_name + " failed to get array of " + varType + " variables names.");
678  }
679 
680  // Then convert from array of C strings to vector of strings, freeing the allocated space as we go
681  for (int i = 0; i < variableCount; ++i) {
682  (*var_names)[i] = std::string(names_array[i]);
683  free(names_array[i]);
684  }
685 
686  return var_names;
687  }
