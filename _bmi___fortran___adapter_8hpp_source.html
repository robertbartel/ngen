380 template<class T>
381 void SetValue(std::string name, std::vector<T> src) {
382 int item_size, total_bytes;
383 try {
384 item_size = GetVarItemsize(name);
385 total_bytes = GetVarNbytes(name);
386 }
387 catch (std::runtime_error &e) {
388 throw std::runtime_error("Cannot set " + name + " variable of " + model_name +
389 "; unable to test item and array sizes are equal (does variable " + name +
390 " exist for model " + model_name + "?)");
391 }
392 if (item_size != sizeof(src[0])) {
393 throw std::runtime_error("Cannot set " + name + " variable of " + model_name +
394 " with values of different item size");
395 }
396 if (src.size() != total_bytes / item_size) {
397 throw std::runtime_error(
398 "Cannot set " + name + " variable of " + model_name + " from vector of size " +
399 std::to_string(src.size) + " (expected size " + std::to_string(total_bytes / item_size) +
400 ")");
401 }
402 SetValue(std::move(name), static_cast<void *>(src.data()));
403 }

687 std::shared_ptr<std::vector<std::string>> inner_get_variable_names(bool is_input_variables) {
688 // Writing this def once here at beginning: fewer lines, but this may or may not be used.
689 std::string varType = (is_input_variables) ? "input" : "output";
690
691 // Obtain this via inner functions, which should use the model directly and not other member functions
692 int variableCount;
693 try {
694 variableCount = (is_input_variables) ? inner_get_input_item_count() : inner_get_output_item_count();
695 }
696 catch (const std::exception &e) {
697 throw std::runtime_error(model_name + " failed to count of " + varType + " variable names array.");
698 }
699
700 // With variable count now obtained, create the vector
701 std::shared_ptr<std::vector<std::string>> var_names = std::make_shared<std::vector<std::string>>(
702 std::vector<std::string>(variableCount));
703
704 // Must get the names from the model as an array of C strings
705 // The array can be on the stack ...
706 char* names_array[variableCount];
707 // ... but allocate the space for the individual C strings (i.e., the char * elements)
708 for (int i = 0; i < variableCount; i++) {
709 names_array[i] = static_cast<char *>(malloc(sizeof(char) * BMI_MAX_VAR_NAME));
710 }
711
712 // With the necessary char** in hand, get the names from the model
713 int names_result;
714 if (is_input_variables) {
715 names_result = get_input_var_names(bmi_model.get(), names_array);
716 }
717 else {
718 names_result = get_output_var_names(bmi_model.get(), names_array);
719 }
720 if (names_result != BMI_SUCCESS) {
721 throw std::runtime_error(model_name + " failed to get array of " + varType + " variables names.");
722 }
723
724 // Then convert from array of C strings to vector of strings, freeing the allocated space as we go
725 for (int i = 0; i < variableCount; ++i) {
726 (*var_names)[i] = std::string(names_array[i]);
727 free(names_array[i]);
728 }
729
730 return var_names;
731 }
