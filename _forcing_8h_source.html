192 size_t get_ts_index_for_time(const time_t &epoch_time) override {
193 if (epoch_time < start_date_time_epoch) {
194 throw std::out_of_range("Forcing had bad pre-start time for index query: " + std::to_string(epoch_time));
195 }
196 size_t i = 0;
197 // 1 hour
198 time_t seconds_in_time_step = 3600;
199 time_t time = start_date_time_epoch;
200 while (epoch_time >= time + seconds_in_time_step && time < end_date_time_epoch) {
201 i++;
202 time += seconds_in_time_step;
203 }
204 // The end_date_time_epoch is the epoch value of the BEGINNING of the last time step, not its end.
205 // I.e., to make sure we cover it, we have to go another time step beyond.
206 if (time >= end_date_time_epoch + 3600) {
207 throw std::out_of_range("Forcing had bad beyond-end time for index query: " + std::to_string(epoch_time));
208 }
209 else {
210 return i;
211 }

226 double get_value(const std::string &output_name, const time_t &init_time, const long &duration_s,
227 const std::string &output_units) override
228 {
229 size_t current_index;
230 long time_remaining = duration_s;
231 try {
232 current_index = get_ts_index_for_time(init_time);
233 }
234 catch (const std::out_of_range &e) {
235 throw std::out_of_range("Forcing had bad init_time " + std::to_string(init_time) + " for value request");
236 }
237
238 std::vector<double> involved_time_step_values;
239 std::vector<long> involved_time_step_seconds;
240 long ts_involved_s;
241
242 time_t first_time_step_start_epoch = start_date_time_epoch + (current_index * 3600);
243 // Handle the first time step differently, since we need to do more to figure out how many seconds came from it
244 // Total time step size minus the offset of the beginning, before the init time
245 ts_involved_s = 3600 - (init_time - first_time_step_start_epoch);
246
247 involved_time_step_seconds.push_back(ts_involved_s);
248 involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
249 time_remaining -= ts_involved_s;
250 current_index++;
251
252 while (time_remaining > 0) {
253 ts_involved_s = time_remaining > 3600 ? 3600 : time_remaining;
254 involved_time_step_seconds.push_back(ts_involved_s);
255 involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
256 time_remaining -= ts_involved_s;
257 current_index++;
258 }
259 double value = 0;
260 for (size_t i = 0; i < involved_time_step_values.size(); ++i) {
261 if (is_param_sum_over_time_step(output_name))
262 value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / 3600.0);
263 else
264 value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / (double)duration_s);
265 }
266 return value;
267 }

276 inline double get_value_for_param_name(const std::string& name, int index) {
277 if (index >= precipitation_rate_meters_per_second_vector.size() || index >= AORC_vector.size()) {
278 throw std::out_of_range("Forcing had bad index " + std::to_string(index) + " for value lookup of " + name);
279 }
280
281 if (name == AORC_FIELD_NAME_PRECIP_RATE) {
282 return precipitation_rate_meters_per_second_vector.at(index);
283 }
284 if (name == AORC_FIELD_NAME_SOLAR_SHORTWAVE || name == CSDMS_STD_NAME_SOLAR_SHORTWAVE) {
285 return AORC_vector.at(index).DSWRF_surface_W_per_meters_squared;
286 }
287 if (name == AORC_FIELD_NAME_SOLAR_LONGWAVE || name == CSDMS_STD_NAME_SOLAR_LONGWAVE) {
288 return AORC_vector.at(index).DLWRF_surface_W_per_meters_squared;
289 }
290 if (name == AORC_FIELD_NAME_PRESSURE_SURFACE || name == CSDMS_STD_NAME_SURFACE_AIR_PRESSURE) {
291 return AORC_vector.at(index).PRES_surface_Pa;
292 }
293 if (name == AORC_FIELD_NAME_TEMP_2M_AG || name == CSDMS_STD_NAME_SURFACE_TEMP) {
294 return AORC_vector.at(index).TMP_2maboveground_K;
295 }
296 if (name == AORC_FIELD_NAME_APCP_SURFACE || name == CSDMS_STD_NAME_LIQUID_EQ_PRECIP_RATE) {
297 return AORC_vector.at(index).APCP_surface_kg_per_meters_squared;
298 }
299 if (name == AORC_FIELD_NAME_WIND_U_10M_AG || name == CSDMS_STD_NAME_WIND_U_X) {
300 return AORC_vector.at(index).UGRD_10maboveground_meters_per_second;
301 }
302 if (name == AORC_FIELD_NAME_WIND_V_10M_AG || name == CSDMS_STD_NAME_WIND_V_Y) {
303 return AORC_vector.at(index).VGRD_10maboveground_meters_per_second;
304 }
305 if (name == AORC_FIELD_NAME_SPEC_HUMID_2M_AG || name == CSDMS_STD_NAME_HUMIDITY) {
306 return AORC_vector.at(index).SPFH_2maboveground_kg_per_kg;
307 }
308 else {
309 throw std::runtime_error("Cannot get forcing value for unrecognized parameter name '" + name + "'.");
310 }

317 inline void check_forcing_vector_index_bounds()
318 {
319 //Check if forcing index is less than zero and if so, set to zero.
320 if (forcing_vector_index < 0)
321 {
322 forcing_vector_index = 0;
324 cout << "WARNING: Forcing vector index is less than zero. Therefore, setting index to zero." << endl;
325 }
326
327 //Check if forcing index is greater than or equal to the size of the size of the precipiation vector and if so, set to zero.
328 else if (forcing_vector_index >= precipitation_rate_meters_per_second_vector.size())
329 {
330 forcing_vector_index = precipitation_rate_meters_per_second_vector.size() - 1;
332 cout << "WARNING: Reached beyond the size of the forcing vector. Therefore, setting index to last value of the vector." << endl;
333 }
334
335 return;

578 void read_forcing(string file_name)
579 {
580 //Call CSVReader constuctor
581 CSVReader reader(file_name);
582
583 //Get the data from CSV File
584 std::vector<std::vector<std::string> > data_list = reader.getData();
585
586 //Iterate through CSV starting on the third row
587 for (int i = 2; i < data_list.size(); i++)
588 {
589 //Row vector
590 std::vector<std::string>& vec = data_list[i];
591
592 //Declare pointer to struct for the current row date-time utc
593 struct tm *current_row_date_time_utc;
594
595 //Allocate memory to struct for the current row date-time utc
596 current_row_date_time_utc = new tm();
597
598 //Year
599 string year_str = vec[0];
600 int year = stoi(year_str);
601 current_row_date_time_utc->tm_year = year - 1900;
602
603 //Month
604 string month_str = vec[1];
605 int month = stoi(month_str);
606 current_row_date_time_utc->tm_mon = month - 1;
607
608 //Day
609 string day_str = vec[2];
610 int day = stoi(day_str);
611 current_row_date_time_utc->tm_mday = day;
612
613 //Hour
614 string hour_str = vec[3];
615 int hour = stoi(hour_str);
616 current_row_date_time_utc->tm_hour = hour;
617
618 //Convert current row date-time utc to epoch time
619 time_t current_row_date_time_epoch = timegm(current_row_date_time_utc);
620
621 //If the current row date-time is within the model date-time range, then add precipitation to vector
622 if (start_date_time_epoch <= current_row_date_time_epoch && current_row_date_time_epoch <= end_date_time_epoch)
623 {
624 //Precipitation
625 string precip_str = vec[5];
626
627 //Convert from string to double and from mm/hr to m/s
628 double precip = boost::lexical_cast<double>(precip_str) / (1000 * 3600);
629
630 //Add precip to vector
631 precipitation_rate_meters_per_second_vector.push_back(precip);
632 }
633
634 //Free memory from struct
635 delete current_row_date_time_utc;
636 }
637 }

644 void read_forcing_aorc(string file_name)
645 {
646 is_aorc_data = true;
647 //Call CSVReader constuctor
648 CSVReader reader(file_name);
649
650 //Get the data from CSV File
651 std::vector<std::vector<std::string> > data_list = reader.getData();
652 //Some quick validation of the data we read
653 if( data_list.size() > 0 ){
654 auto header = data_list[0];
655 if( header.size() != 10 ){
656 throw std::runtime_error("Error: Forcing data " + file_name +
657 " does not have the expected number of columns. Did you mean to use "+
658 "csvPerFeature forcing provider instead?");
659 }
660 }
661 else {
662 throw std::runtime_error("Error: Forcing data " + file_name +
663 " does not have readable rows.");
664 }
665
666 //Iterate through CSV starting on the second row
667 for (int i = 1; i < data_list.size(); i++)
668 {
669 //Row vector
670 std::vector<std::string>& vec = data_list[i];
671
672 //Declare struct for the current row date-time
673 struct tm current_row_date_time_utc;
674
675 //Allocate memory to struct for the current row date-time
676 current_row_date_time_utc = tm();
677
678 //Grab time string from first column
679 string time_str = vec[0];
680
681 //Convert time string to time struct
682 strptime(time_str.c_str(), "%Y-%m-%d %H:%M:%S", &current_row_date_time_utc);
683
684 //Convert current row date-time UTC to epoch time
685 time_t current_row_date_time_epoch = timegm(&current_row_date_time_utc);
686
687 //Ensure that forcing data covers the entire model period. Otherwise, throw an error.
688 if (i == 1 && start_date_time_epoch < current_row_date_time_epoch)
689 {
691 //cout << "WARNING: Forcing data begins after the model start time." << endl;
692 struct tm start_date_tm;
693 gmtime_r(&start_date_time_epoch, &start_date_tm);
694
695 char tm_buff[128];
696 strftime(tm_buff, 128, "%Y-%m-%d %H:%M:%S", &start_date_tm);
697 throw std::runtime_error("Error: Forcing data " + file_name + " begins after the model start time:" + std::string(tm_buff) + " < " + time_str);
698 }
699
700
701 else if (i == data_list.size() - 1 && current_row_date_time_epoch < end_date_time_epoch)
703 cout << "WARNING: Forcing data ends before the model end time." << endl;
704 //throw std::runtime_error("Error: Forcing data ends before the model end time.");
705
706 //If the current row date-time is within the model date-time range, then add precipitation to vector
707 if (start_date_time_epoch <= current_row_date_time_epoch && current_row_date_time_epoch <= end_date_time_epoch)
708 {
709 //Grab data from columns
710 string APCP_surface_str = vec[1];
711 string DLWRF_surface_str = vec[2];
712 string DSWRF_surface_str = vec[3];
713 string PRES_surface_str = vec[4];
714 string SPFH_2maboveground_str = vec[5];
715 string TMP_2maboveground_str = vec[6];
716 string UGRD_10maboveground_str = vec[7];
717 string VGRD_10maboveground_str = vec[8];
718 string precip_rate_str = vec[9];
719
720 //Declare AORC struct
721 AORC_data AORC;
722
723 //Convert from strings to doubles and add to AORC struct
724 AORC.APCP_surface_kg_per_meters_squared = boost::lexical_cast<double>(APCP_surface_str);
725 AORC.DLWRF_surface_W_per_meters_squared = boost::lexical_cast<double>(DLWRF_surface_str);
726 AORC.DSWRF_surface_W_per_meters_squared = boost::lexical_cast<double>(DSWRF_surface_str);
727 AORC.PRES_surface_Pa = boost::lexical_cast<double>(PRES_surface_str);
728 AORC.SPFH_2maboveground_kg_per_kg = boost::lexical_cast<double>(SPFH_2maboveground_str);
729 AORC.TMP_2maboveground_K = boost::lexical_cast<double>(TMP_2maboveground_str);
730 AORC.UGRD_10maboveground_meters_per_second = boost::lexical_cast<double>(UGRD_10maboveground_str);
731 AORC.VGRD_10maboveground_meters_per_second = boost::lexical_cast<double>(VGRD_10maboveground_str);
732
733
734 //Add AORC struct to AORC vector
735 AORC_vector.push_back(AORC);
736
737 //Convert precip_rate from string to double
738 double precip_rate = boost::lexical_cast<double>(precip_rate_str);
739
740 //Add data to vectors
741 precipitation_rate_meters_per_second_vector.push_back(precip_rate);
742 time_epoch_vector.push_back(current_row_date_time_epoch);
743 }
744
745 //Free memory from struct
746 //delete current_row_date_time_utc;
747 }
