30  static coordinate_t build_point(boost::property_tree::ptree &tree) {
31  std::vector<double> point;
32 
33  // GeoJSON specifies that the details of the point will lie in a 1D array under "coordinates"
34  for(auto &shape : tree.get_child("coordinates")) {
35  // The specification of the points will lie within string values, so we need to convert the strings
36  // to double values
37  point.push_back(std::stod(shape.second.data()));
38  }
39 
40  // The specification of GeoJson specifies that points will be described within a 1D array of length 2
41  // (the longitude, then the latitude). Use both in order to construct the point and return it
42  return coordinate_t(point[0], point[1]);
43  }

377  static GeoJSON build_collection(const boost::property_tree::ptree tree, const std::vector<std::string> ids={}) {
378  std::vector<double> bbox_values;
379  std::vector<Feature> features;
380  PropertyMap foreign_members;
381  std::string tmp_id; //a temporary string to hold feature identities
382 
383  for (auto& child : tree) {
384  if (child.first == "bbox") {
385  std::vector<std::string> bounding_box;
386  for (auto &feature : tree.get_child("bbox")) {
387  bounding_box.push_back(feature.second.data());
388  }
389 
390  for (int point_index = 0; point_index < bounding_box.size(); point_index++) {
391  bbox_values.push_back(std::stod(bounding_box[point_index]));
392  }
393  }
394  else if (child.first == "features") {
395  boost::optional<const boost::property_tree::ptree&> e = tree.get_child_optional("features");
396 
397  if (e) {
398  for(auto feature_tree : *e) {
399  Feature feature = build_feature(feature_tree.second);
400  tmp_id = feature->get_id();
401  //TODO feature identity isn't 100% spec compliant. GeoJSON allows for a feature to have an
402  //optional id, but the input files set id under the 'property' key, so when a feature is constructed
403  //feature->get_id() returns '' because the feature itself doesn't have an id, so we hae to read
404  //from the associated properties to find the id. If the inputs change, we will need to adjust
405  //this line to read feature->get_id()
406  if( tmp_id == "" ) {
407  try {
408  tmp_id = feature->get_property("id").as_string();
409  feature->set_id(tmp_id);
410  }
411  catch (const std::out_of_range& error) {
412  tmp_id = "";
413  }
414  }
415 
416  if( ids.empty() || std::find(ids.begin(), ids.end(), tmp_id) != ids.end() ) {
417  //ids is empty, meaning we want all features,
418  //or feature id was found in the provided ids vector
419  //so hold the feature to add to collection later
420 
421  features.push_back(feature);
422  }
423  }
424  }
425  else {
426  std::cout << "No features were found" << std::endl;
427  }
428  }
429  else {
430  foreign_members.emplace(child.first, JSONProperty(child.first, child.second));
431  }
432  }
433 
434  GeoJSON collection = std::make_shared<FeatureCollection>(FeatureCollection(features, bbox_values));
435 
436  for (Feature feature : features) {
437  if (feature->get_id() != "") {
438  collection->add_feature_id(feature->get_id(), feature);
439  }
440  }
441 
442  return collection;
443  }
