555  void inner_create_formulation(geojson::PropertyMap properties, bool needs_param_validation) {
556  if (needs_param_validation) {
557  validate_parameters(properties);
558  }
559  // Required parameters first
560  set_bmi_init_config(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG).as_string());
561  set_bmi_main_output_var(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__MAIN_OUT_VAR).as_string());
562  set_model_type_name(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__MODEL_TYPE).as_string());
563  set_bmi_using_forcing_file(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__USES_FORCINGS).as_boolean());
564 
565  // Then optional ...
566 
567  // Note that this must be present if bmi_using_forcing_file is true
568  if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__FORCING_FILE) != properties.end())
569  set_forcing_file_path(properties.at(BMI_REALIZATION_CFG_PARAM_OPT__FORCING_FILE).as_string());
570  else if (is_bmi_using_forcing_file())
571  throw std::runtime_error("Can't create BMI formulation: using_forcing_file true but no file path set");
572 
573  if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__ALLOW_EXCEED_END) != properties.end()) {
574  set_allow_model_exceed_end_time(
575  properties.at(BMI_REALIZATION_CFG_PARAM_OPT__ALLOW_EXCEED_END).as_boolean());
576  }
577  if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__FIXED_TIME_STEP) != properties.end()) {
578  set_bmi_model_time_step_fixed(
579  properties.at(BMI_REALIZATION_CFG_PARAM_OPT__FIXED_TIME_STEP).as_boolean());
580  }
581 
582  auto std_names_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__VAR_STD_NAMES);
583  if (std_names_it != properties.end()) {
584  geojson::PropertyMap names_map = std_names_it->second.get_values();
585  for (auto& names_it : names_map) {
586  bmi_var_names_map.insert(
587  std::pair<std::string, std::string>(names_it.first, names_it.second.as_string()));
588  }
589  }
590 
591  // Do this next, since after checking whether other input variables are present in the properties, we can
592  // now construct the adapter and init the model
593  set_bmi_model(construct_model(properties));
594 
595  // Make sure that this is able to interpret model time and convert to real time, since BMI model time is
596  // usually starting at 0 and just counting up
597  determine_model_time_offset();
598 
599  // Output variable subset and order, if present
600  auto out_var_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__OUT_VARS);
601  if (out_var_it != properties.end()) {
602  std::vector<geojson::JSONProperty> out_vars_json_list = out_var_it->second.as_list();
603  std::vector<std::string> out_vars(out_vars_json_list.size());
604  for (int i = 0; i < out_vars_json_list.size(); ++i) {
605  out_vars[i] = out_vars_json_list[i].as_string();
606  }
607  set_output_variable_names(out_vars);
608  }
609  // Otherwise, just take what literally is provided by the model
610  else {
611  set_output_variable_names(get_bmi_model()->GetOutputVarNames());
612  }
613 
614  // Output header fields, if present
615  auto out_headers_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__OUT_HEADER_FIELDS);
616  if (out_headers_it != properties.end()) {
617  std::vector<geojson::JSONProperty> out_headers_json_list = out_var_it->second.as_list();
618  std::vector<std::string> out_headers(out_headers_json_list.size());
619  for (int i = 0; i < out_headers_json_list.size(); ++i) {
620  out_headers[i] = out_headers_json_list[i].as_string();
621  }
622  set_output_header_fields(out_headers);
623  }
624  else {
625  set_output_header_fields(get_output_variable_names());
626  }
627  // Create a reference to this for ET by using a WrappedForcingProvider
628  std::shared_ptr<forcing::ForcingProvider> self = std::make_shared<forcing::WrappedForcingProvider>(this);
629  input_forcing_providers[NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP] = self;
630  input_forcing_providers[CSDMS_STD_NAME_POTENTIAL_ET] = self;
631 
632  // Finally, make sure this is set
633  model_initialized = get_bmi_model()->is_model_initialized();
634  }
