702 void inner_create_formulation(geojson::PropertyMap properties, bool needs_param_validation) {
703 if (needs_param_validation) {
704 validate_parameters(properties);
705 }
706 // Required parameters first
707 set_bmi_init_config(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG).as_string());
708 set_bmi_main_output_var(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__MAIN_OUT_VAR).as_string());
709 set_model_type_name(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__MODEL_TYPE).as_string());
710 set_bmi_using_forcing_file(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__USES_FORCINGS).as_boolean());
711
712 // Then optional ...
713
714 // Note that this must be present if bmi_using_forcing_file is true
715 if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__FORCING_FILE) != properties.end())
716 set_forcing_file_path(properties.at(BMI_REALIZATION_CFG_PARAM_OPT__FORCING_FILE).as_string());
717 else if (is_bmi_using_forcing_file())
718 throw std::runtime_error("Can't create BMI formulation: using_forcing_file true but no file path set");
