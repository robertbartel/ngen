495  inline void get_forcing_data_ts_contributions(time_step_t t_delta, const double &model_initial_time,
496  const std::vector<std::string> &params,
497  const std::vector<bool> &is_aorc_param,
498  const std::vector<std::string> &param_units,
499  std::vector<double> &summed_contributions)
500  {
501  // TODO: probably deprecated and needs to be removed
502  time_t model_ts_start_offset, model_ts_seconds_contained_in_forcing_ts, model_epoch_time_s;
503  // Keep track of how much of the model ts delta has not yet had its contribution pulled from some forcing ts
504  time_step_t contribution_seconds_remaining = t_delta;
505 
506  model_epoch_time_s = (time_t) (get_bmi_model()->convert_model_time_to_seconds(model_initial_time)) +
507  get_bmi_model_start_time_forcing_offset_s();
508  // Make sure the forcings are properly primed to have the correct corresponding forcing ts data available
509  while (model_epoch_time_s > forcing.get_time_epoch() + forcing.get_time_step_size()) {
510  forcing.get_next_hourly_precipitation_meters_per_second();
511  }
512  // Though this is a problem at the moment, because we can't go through forcings in reverse order
513  if (model_epoch_time_s < forcing.get_time_epoch()) {
514  // TODO: think if there is a better way to address this
515  // TODO: should we include <typeinfo> and use typeid(this).name() to print class?
516  // (Also, should there be a directive-controlled option for that?)
517  throw std::runtime_error(get_formulation_type() + " formulation for model '" + get_model_type_name() +
518  "' can't get contributions for model time step " + to_string(t_delta) + " starting at "
519  + to_string(model_epoch_time_s) + ", as current forcing time step doesn't start until "
520  + to_string(forcing.get_time_epoch()));
521  }
522 
523  model_ts_start_offset = model_epoch_time_s - forcing.get_time_epoch();
524 
525  while (contribution_seconds_remaining > 0) {
526  // Note that model_ts_start_offset shift is cleared (set to 0) at end of loop, so only has effect once.
527  // If remainder of model ts (after shift in first iteration) ends before forcing ts ...
528  if ((time_t) contribution_seconds_remaining + model_ts_start_offset < forcing.get_time_step_size()) {
529  model_ts_seconds_contained_in_forcing_ts = (time_t) contribution_seconds_remaining;
530  }
531  else {
532  model_ts_seconds_contained_in_forcing_ts = forcing.get_time_step_size() - model_ts_start_offset;
533  }
534 
535  // Get the contributions from this forcing ts for all the forcing params needed.
536  for (size_t i = 0; i < params.size(); ++i) {
537  // Skip indices for parameters that are not forcings
538  if (!is_aorc_param[i]) {
539  continue;
540  }
541  // This is proportional to ratio of (model ts secs in this forcing ts) to (FORCING ts total secs)
542  if (forcing.is_param_sum_over_time_step(params[i])) {
543  summed_contributions[i] +=
544  forcing.get_converted_value_for_param_in_units(params[i], param_units[i]) *
545  (double) model_ts_seconds_contained_in_forcing_ts /
546  (double) forcing.get_time_step_size();
547  }
548  // This is proportional to the ratio of (model ts secs in this forcing ts) to (MODEL ts total secs)
549  else {
550  summed_contributions[i] +=
551  forcing.get_converted_value_for_param_in_units(params[i], param_units[i]) *
552  (double) model_ts_seconds_contained_in_forcing_ts /
553  (double) t_delta;
554  }
555  }
556 
557  // Account for the processed model time compared to the entire delta
558  contribution_seconds_remaining -= (time_step_t)model_ts_seconds_contained_in_forcing_ts;
559  // The offset should only possibly be non-zero the first time (after, if the model ts extends beyond the first
560  // forcing ts, it always picks up at the beginning of the subsequent forcing ts), so set to 0 now.
561  model_ts_start_offset = 0;
562  // Also, when appropriate incrementing the forcing ts
563  if (contribution_seconds_remaining > 0) {
564  forcing.get_next_hourly_precipitation_meters_per_second();
565  }
566  }
567  }

665  // usually starting at 0 and just counting up
666  determine_model_time_offset();
667 
668  // Output variable subset and order, if present
669  auto out_var_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__OUT_VARS);
670  if (out_var_it != properties.end()) {
671  std::vector<geojson::JSONProperty> out_vars_json_list = out_var_it->second.as_list();
