363 if (std::find(forcing_outputs.begin(), forcing_outputs.end(), output_name) == forcing_outputs.end()) {
364 throw runtime_error(get_formulation_type() + " received invalid output forcing name " + output_name);
365 }
366 // TODO: do this, or something better, later; right now, just assume anything using this as a provider is
367 // consistent with times

377 get_bmi_output_var_name(output_name, bmi_var_name);

379 if( !bmi_var_name.empty() )
380 {
381 //Get forcing value from BMI variable
382 double value = get_var_value_as_double(bmi_var_name);
383

604 void inner_create_formulation(geojson::PropertyMap properties, bool needs_param_validation) {
605 if (needs_param_validation) {
606 validate_parameters(properties);
607 }
608 // Required parameters first
609 set_bmi_init_config(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG).as_string());
610 set_bmi_main_output_var(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__MAIN_OUT_VAR).as_string());
611 set_model_type_name(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__MODEL_TYPE).as_string());
612 set_bmi_using_forcing_file(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__USES_FORCINGS).as_boolean());
613
614 // Then optional ...
615
616 // Note that this must be present if bmi_using_forcing_file is true
617 if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__FORCING_FILE) != properties.end())
618 set_forcing_file_path(properties.at(BMI_REALIZATION_CFG_PARAM_OPT__FORCING_FILE).as_string());
619 else if (is_bmi_using_forcing_file())
620 throw std::runtime_error("Can't create BMI formulation: using_forcing_file true but no file path set");
621
622 if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__ALLOW_EXCEED_END) != properties.end()) {
623 set_allow_model_exceed_end_time(
624 properties.at(BMI_REALIZATION_CFG_PARAM_OPT__ALLOW_EXCEED_END).as_boolean());
625 }
626 if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__FIXED_TIME_STEP) != properties.end()) {
627 set_bmi_model_time_step_fixed(
628 properties.at(BMI_REALIZATION_CFG_PARAM_OPT__FIXED_TIME_STEP).as_boolean());
629 }
630
631 auto std_names_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__VAR_STD_NAMES);
632 if (std_names_it != properties.end()) {
633 geojson::PropertyMap names_map = std_names_it->second.get_values();
634 for (auto& names_it : names_map) {
635 bmi_var_names_map.insert(
636 std::pair<std::string, std::string>(names_it.first, names_it.second.as_string()));
637 }
638 }
639
640 // Do this next, since after checking whether other input variables are present in the properties, we can
641 // now construct the adapter and init the model
642 set_bmi_model(construct_model(properties));
643
644 // Make sure that this is able to interpret model time and convert to real time, since BMI model time is
645 // usually starting at 0 and just counting up
646 determine_model_time_offset();
647
648 // Output variable subset and order, if present
649 auto out_var_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__OUT_VARS);
650 if (out_var_it != properties.end()) {
651 std::vector<geojson::JSONProperty> out_vars_json_list = out_var_it->second.as_list();
652 std::vector<std::string> out_vars(out_vars_json_list.size());
653 for (int i = 0; i < out_vars_json_list.size(); ++i) {
654 out_vars[i] = out_vars_json_list[i].as_string();
655 }
656 set_output_variable_names(out_vars);
657 }
658 // Otherwise, just take what literally is provided by the model
659 else {
660 set_output_variable_names(get_bmi_model()->GetOutputVarNames());
661 }
662
663 // Output header fields, if present
664 auto out_headers_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__OUT_HEADER_FIELDS);
665 if (out_headers_it != properties.end()) {
666 std::vector<geojson::JSONProperty> out_headers_json_list = out_var_it->second.as_list();
667 std::vector<std::string> out_headers(out_headers_json_list.size());
668 for (int i = 0; i < out_headers_json_list.size(); ++i) {
669 out_headers[i] = out_headers_json_list[i].as_string();
670 }
671 set_output_header_fields(out_headers);
672 }
673 else {
674 set_output_header_fields(get_output_variable_names());
675 }
676 // Create a reference to this for ET by using a WrappedForcingProvider
677 std::shared_ptr<forcing::ForcingProvider> self = std::make_shared<forcing::WrappedForcingProvider>(this);
678 input_forcing_providers[NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP] = self;
679 input_forcing_providers[CSDMS_STD_NAME_POTENTIAL_ET] = self;
680
681 // Output precision, if present
682 auto out_precision_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__OUTPUT_PRECISION);
683 if (out_precision_it != properties.end()) {
684 set_output_precision(properties.at(BMI_REALIZATION_CFG_PARAM_OPT__OUTPUT_PRECISION).as_natural_number());
