330 forcing_params get_global_forcing_params(std::string identifier, simulation_time_params &simulation_time_config) {
331 std::string path = this->global_forcing.at("path").as_string();
332 std::string provider = "";
333 if(this->global_forcing.count("provider") != 0){
334 provider = global_forcing.at("provider").as_string();
335 }
336 if (this->global_forcing.count("file_pattern") == 0) {
337 return forcing_params(
338 path,
339 provider,
340 simulation_time_config.start_time,
341 simulation_time_config.end_time
342 );
343 }
344
345 // Since we are given a pattern, we need to identify the directory and pull out anything that matches the pattern
346 if (path.compare(path.size() - 1, 1, "/") != 0) {
347 path += "/";
348 }
349
350 std::string filepattern = this->global_forcing.at("file_pattern").as_string();
351
352 int id_index = filepattern.find("{{id}}");
353
354 // If an index for '{{id}}' was found, we can count on that being where the id for this realization can be found.
355 // For instance, if we have a pattern of '.*{{id}}_14_15.csv' and this is named 'cat-87',
356 // this will match on 'stuff_example_cat-87_14_15.csv'
357 if (id_index != std::string::npos) {
358 filepattern = filepattern.replace(id_index, sizeof("{{id}}") - 1, identifier);
359 }
360
361 // Create a regular expression used to identify proper file names
362 std::regex pattern(filepattern);
363
364 // A stream providing the functions necessary for evaluating a directory:
365 // https://www.gnu.org/software/libc/manual/html_node/Opening-a-Directory.html#Opening-a-Directory
366 DIR *directory = nullptr;
367
368 // structure representing the member of a directory: https://www.gnu.org/software/libc/manual/html_node/Directory-Entries.html
369 struct dirent *entry = nullptr;
370
371 // Attempt to open the directory for evaluation
372 directory = opendir(path.c_str());
373 // Allow for a few retries in certain failure situations
374 size_t attemptCount = 0;
375 std::string errMsg;
376 while (directory == nullptr && attemptCount++ < 5) {
377 // For several error codes, we should break immediately and not retry
378 if (errno == ENOENT) {
379 errMsg = "No such file or directory.";
380 break;
381 }
382 if (errno == ENXIO) {
383 errMsg = "No such device or address.";
384 break;
385 }
386 if (errno == EACCES) {
387 errMsg = "Permission denied.";
388 break;
389 }
390 if (errno == EPERM) {
391 errMsg = "Operation not permitted.";
392 break;
393 }
394 if (errno == ENOTDIR) {
395 errMsg = "File at provided path is not a directory.";
396 break;
397 }
398 if (errno == EMFILE) {
399 errMsg = "The current process has too many open files.";
400 break;
401 }
402 if (errno == ENFILE) {
403 errMsg = "The system has too many open files.";
404 break;
405 }
406 sleep(2);
407 directory = opendir(path.c_str());
408 errMsg = "Received system error number " + std::to_string(errno);
409 }
410
411 // If the directory could be found and opened, we can go ahead and iterate
412 if (directory != nullptr) {
413 while ((entry = readdir(directory))) {
414 // If the entry is a regular file or symlink AND the name matches the pattern,
415 // we can consider this ready to be interpretted as valid forcing data (even if it isn't)
416 if ((entry->d_type == DT_REG or entry->d_type == DT_LNK) and std::regex_match(entry->d_name, pattern)) {
417 return forcing_params(
418 path + entry->d_name,
419 provider,
420 simulation_time_config.start_time,
421 simulation_time_config.end_time
422 );
423 }
424 }
425 }
426 else {
427 // The directory wasn't found or otherwise couldn't be opened; forcing data cannot be retrieved
428 throw std::runtime_error("Error opening forcing data dir '" + path + "' after " + std::to_string(attemptCount) + " attempts: " + errMsg);
429 }
430
431 closedir(directory);
432
433 throw std::runtime_error("Forcing data could not be found for '" + identifier + "'");
434 }
