36  JSONProperty(std::string value_key, const boost::property_tree::ptree& property_tree) {
37  key = value_key;
38 
39  if (property_tree.empty() && !property_tree.data().empty()) {
40  // This is a terminal node and has a raw value
41  // TODO: Add handling for nested objects by determining if property.second is another ptree
42  std::string value = property_tree.data();
43 
44  if (value == "true" || value == "false") {
45  type = PropertyType::Boolean;
46  boolean = value == "true";
47  }
48  else {
49  bool is_numeric = true;
50  bool is_real = true;
51  bool decimal_already_hit = false;
52 
53  for(int character_index = 0; character_index < value.length(); character_index++) {
54  char character = value[character_index];
55 
56  // If the first character is a '0' or isn't a digit, the whole value cannot be a number
57  if (character_index == 0 && character == '0') {
58  is_numeric = false;
59  }
60  else if (character != '.' && !std::isdigit(character)) {
61  // If this character isn't a decimal point and isn't a digit, the whole value cannot be a number
62  is_numeric = false;
63  is_real = false;
64  break;
65  }
66  else if (character == '.' && decimal_already_hit) {
67  // If this character is a decimal point, but we've already seen one, the whole value cannot be a number
68  is_real = false;
69  break;
70  }
71  else if (character == '.') {
72  // If a decimal point is seen, the whole value cannot be an integer
73  is_numeric = false;
74  decimal_already_hit = true;
75  }
76  }
77 
78  // If the value can be represented as a whole number, we want to go with that
79  if (is_numeric) {
80  type = PropertyType::Natural;
81  natural_number = std::stol(value);
82  }
83  else if (is_real) {
84  type = PropertyType::Real;
85  real_number = std::stod(value);
86  }
87  else {
88  // Otherwise we'll store everything as a raw string
89  type = PropertyType::String;
90  string = value;
91  }
92  }
93  }
94  else {
95  // This isn't a terminal node, therefore represents an object
96  type = PropertyType::Object;
97  for (auto &property : property_tree) {
98  values.emplace(property.first, JSONProperty(property.first, property.second));
99  }
100  }
101  }
