85  void Initialize() {
86  // If there was previous init attempt but w/ failure exception, throw runtime error and include previous
87  // message
88  if (model_initialized && !init_exception_msg.empty()) {
89  throw runtime_error(

91  }
92  // If there was previous init attempt w/ (implicitly) no exception on previous attempt, just return
93  else if (model_initialized) {
94  return;
95  }
96  else if (!utils::FileChecker::file_is_readable(bmi_init_config)) {
97  init_exception_msg = "Cannot initialize " + model_name + " using unreadable file '"
98  + bmi_init_config + "'";
99  throw runtime_error(init_exception_msg);
100  }
101  else {
102  try {
103  // TODO: make this same name as used with other testing (adjust name in docstring above also)
104  construct_and_init_backing_model();

110  catch (exception& e) {
