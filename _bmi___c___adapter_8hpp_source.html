495  inline void execModuleRegistration() {
496  if (get_dyn_lib_handle() == nullptr) {
497  dynamic_library_load();
498  }
499  void *symbol;
500  C_Bmi *(*dynamic_register_bmi)(C_Bmi *model);
501 
502  try {
503  // Acquire the BMI struct func pointer registration function
504  symbol = dynamic_load_symbol(get_bmi_registration_function());
505  dynamic_register_bmi = (C_Bmi *(*)(C_Bmi *)) symbol;
506  // Call registration function, which (for C libs) sets up object's pointed-to member BMI struct
507  dynamic_register_bmi(bmi_model.get());
508  }
509  catch (const ::external::ExternalIntegrationException &e) {
510  // "Override" the default message in this case
511  this->init_exception_msg =
512  "Cannot init " + this->model_name + " without valid library registration function: " +
513  this->init_exception_msg;
514  throw ::external::ExternalIntegrationException(this->init_exception_msg);
515  }
