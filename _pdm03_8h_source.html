58  inline void Pdm03(int model_time_step,
59  double maximum_combined_contents,
60  double scaled_distribution_fn_shape_parameter,
61  double *final_height_reservoir,
62  double max_height_soil_moisture_storerage_tank,
63  double *total_effective_rainfall,
64  double *actual_et,
65  double *final_storage_upper_zone,
66  double precipitation,
67  double potential_et,
68  double vegetation_adjustment)
69  {
70  /* local variables */
71  //double Cbeg, OV2, PPinf, Hint, Cint, OV1;
72  double cbeg, ov2, ppinf, hint, cint, ov1;
73 
74  //Storage contents at begining
75  //Cbeg = Cpar * (1.0 - pow(1.0-(*XHuz/Huz),1.0+B));
76  cbeg = maximum_combined_contents * (1.0 - pow(1.0-(*final_height_reservoir/max_height_soil_moisture_storerage_tank),
77  1.0+scaled_distribution_fn_shape_parameter));
78 
79  //Compute effective rainfall filling all storage elements
80  //OV2 = max(0.0, effPrecip + *XHuz - Huz);
81  ov2 = max(0.0, precipitation + *final_height_reservoir - max_height_soil_moisture_storerage_tank);
82 
83  //Remaining net rainfall
84  //PPinf = effPrecip - OV2;
85  ppinf = precipitation - ov2;
86 
87  //New actual height
88  //Hint = min(Huz, *XHuz+PPinf);
89  hint = min(max_height_soil_moisture_storerage_tank, *final_height_reservoir+ppinf);
90 
91  //New storage content
92  //Cint = Cpar*(1.0-pow(1.0-(Hint/Huz),1.0+B));
93  cint = maximum_combined_contents*(1.0-pow(1.0-(hint/max_height_soil_moisture_storerage_tank),
94  1.0+scaled_distribution_fn_shape_parameter));
95 
96  //Additional effective rainfall produced by stores smaller than Cmax
97  //OV1 = max(0.0, PPinf + Cbeg - Cint);
98  ov1 = max(0.0, ppinf + cbeg - cint);
99 
100  //Compute total effective rainfall
101  //*OV = OV1 + OV2;
102  *total_effective_rainfall = ov1 + ov2;
103 
104  //Compute actual evapotranspiration
105  //*AE = min(Cint,(Cint/Cpar)*PE*Kv);
106  *actual_et = min(cint,(cint/maximum_combined_contents)*potential_et*vegetation_adjustment);
107 
108  //Storage contents after ET
109  //*XCuz = max(0.0,Cint - *AE);
110  *final_storage_upper_zone = max(0.0,cint - *actual_et);
111 
112  //Compute final height of the reservoir
113  //*XHuz = Huz*(1.0-pow(1.0-(*XCuz/Cpar),1.0/(1.0+B)));
114  *final_height_reservoir = max_height_soil_moisture_storerage_tank*
115  (1.0-pow(1.0-(*final_storage_upper_zone/maximum_combined_contents),
116  1.0/(1.0+scaled_distribution_fn_shape_parameter)));
117 
118  return;

150  inline double calculate_hamon_pet(double average_temp_degree_celcius,
151  double latitude_of_catchment_centoid_degree,
152  int day_of_year)
153  {
154  /* local variables
155  solar_declination - [radian]
156  day_length - [hours]
157  saturation_vapor_pressure - [kPa]
158  potential_et - potential evapotranspiration this time step [mm/day]
159  */
160  double solar_declination, day_length, saturation_vapor_pressure, potential_et;
161 
162  solar_declination =
163  asin(0.39795*cos(0.2163108 + 2.0 * atan(0.9671396*tan(0.00860*
164  (double)(day_of_year-186)))));
165  day_length = 24.0 - (24.0/M_PI)*(acos((sin(0.8333*M_PI/180.0)+
166  sin(latitude_of_catchment_centoid_degree*M_PI/180.0)*
167  sin(solar_declination))/
168  (cos(latitude_of_catchment_centoid_degree*M_PI/180.0)*
169  cos(solar_declination))));
170  saturation_vapor_pressure = 0.6108*exp((17.27*average_temp_degree_celcius)/(237.3+average_temp_degree_celcius));
171  potential_et = (715.5*day_length/24.0)*saturation_vapor_pressure/(average_temp_degree_celcius+ 273.2);
172 
173  return potential_et;

196  inline double greg_2_jul(long year, long mon, long day, long h, long mi, double se)
197  {
198  long m = mon, d = day, y = year;
199  long c, ya, j;
200  double seconds = h * 3600.0 + mi * 60 + se;
201 
202  if (m > 2)
203  m -= 3;
204  else {
205  m += 9;
206  --y;
207  }
208  c = y / 100L;
209  ya = y - (100L * c);
210  j = (146097L * c) / 4L + (1461L * ya) / 4L + (153L * m + 2L) / 5L + d + 1721119L;
211  if (seconds < 12 * 3600.0) {
212  j--;
213  seconds += 12.0 * 3600.0;
214  }
215  else {
216  seconds = seconds - 12.0 * 3600.0;
217  }
218  return (j + (seconds / 3600.0) / 24.0);
219  }

221  /* Julian date converter. Takes a julian date (the number of days since
222  ** some distant epoch or other), and returns an int pointer to static space.
223  ** ip[0] = month;
224  ** ip[1] = day of month;
225  ** ip[2] = year (actual year, like 1977, not 77 unless it was 77 a.d.);
226  ** ip[3] = day of week (0->Sunday to 6->Saturday)
227  ** These are Gregorian.
228  ** Copied from Algorithm 199 in Collected algorithms of the CACM
229  ** Author: Robert G. Tantzen, Translator: Nat Howard
230  **
231  **
232  */
233  inline void calc_date(double jd, long *y, long *m, long *d, long *h, long *mi,
234  double *sec)
235  {
236  static int ret[4];
237 
238  long j;
239  double tmp;
240  double frac;
241 
242  j=(long)jd;
243  frac = jd - j;
244 
245  if (frac >= 0.5) {
246  frac = frac - 0.5;
247  j++;
248  }
249  else {
250  frac = frac + 0.5;
251  }
252 
253  ret[3] = (j + 1L) % 7L;
254  j -= 1721119L;
255  *y = (4L * j - 1L) / 146097L;
256  j = 4L * j - 1L - 146097L * *y;
257  *d = j / 4L;
258  j = (4L * *d + 3L) / 1461L;
259  *d = 4L * *d + 3L - 1461L * j;
260  *d = (*d + 4L) / 4L;
261  *m = (5L * *d - 3L) / 153L;
262  *d = 5L * *d - 3 - 153L * *m;
263  *d = (*d + 5L) / 5L;
264  *y = 100L * *y + j;
265  if (*m < 10)
266  *m += 3;
267  else {
268  *m -= 9;
269  *y=*y+1;
270  }
271 
272  tmp = 3600.0 * (frac * 24.0);
273  *h = (long) (tmp / 3600.0);
274  tmp = tmp - *h * 3600.0;
275  *mi = (long) (tmp / 60.0);
276  *sec = tmp - *mi * 60.0;
277  }
