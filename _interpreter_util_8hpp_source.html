76 void addToPath(const std::string &directoryPath) {
77 std::tuple<py::list, std::vector<std::string>> sysPathTuple = getSystemPath();
78 py::list sys_path = std::get<0>(sysPathTuple);
79 std::vector<std::string> sys_path_vector = std::get<1>(sysPathTuple);
80
81 py::object requestedDirPath = Path(directoryPath);
82 if (py::bool_(requestedDirPath.attr("is_dir")())) {
83 if (std::find(sys_path_vector.begin(), sys_path_vector.end(), directoryPath) == sys_path_vector.end()) {
84 sys_path.attr("insert")(sys_path_vector.size(), py::str(directoryPath));
85 }
86 }
87 else {
88 std::string dirPath = py::str(requestedDirPath);
89 throw std::runtime_error("Cannot add non-existing directory '" + dirPath + "' to Python PATH");
90 }
91 }

152 py::module_ getModule(const std::vector<std::string> &moduleLevelNames) {
153 // Start with top-level module name, then descend through attributes as needed
154 py::module_ module = getModule(moduleLevelNames[0]);
155 for (size_t i = 1; i < moduleLevelNames.size(); ++i) {
156 module = module.attr(moduleLevelNames[i].c_str());

173 py::object searchForVenvDir() {
174 py::object current_dir = Path.attr("cwd")();
175 std::vector<py::object> parent_options = {
176 current_dir,
177 current_dir.attr("parent"),
178 current_dir.attr("parent").attr("parent")
179 };
180 std::vector<std::string> dir_name_options = {".venv", "venv"};
181 for (py::object &parent_option : parent_options) {
182 for (const std::string &dir_name_option : dir_name_options) {
183 py::object venv_dir_candidate = parent_option.attr("joinpath")(dir_name_option);
184 if (py::bool_(venv_dir_candidate.attr("is_dir")())) {
185 return venv_dir_candidate;
186 }
187 }
188 }
189 return py::none();
190 }

197 py::list getVenvPackagesDirOptions() {
198 // Look for a local virtual environment directory also, if there is one
199 const char* env_var_venv = std::getenv("VIRTUAL_ENV");
200 py::object venv_dir = env_var_venv != nullptr ? Path(env_var_venv): searchForVenvDir();
201
202 if (!venv_dir.is_none() && py::bool_(venv_dir.attr("is_dir")())) {
203 // Resolve the full path
204 venv_dir = venv_dir.attr("resolve")();
205 // Get options for the packages dir
206 py::list site_packages_options = py::list(venv_dir.attr("glob")("**/site-packages/"));
207 return site_packages_options;
208 }
209
210 return py::list();
211 }

219 inline std::tuple<py::list, std::vector<std::string>> getSystemPath() {
220 py::list sys_path = importedTopLevelModules.find("sys")->second.attr("path");
221 std::vector<std::string> sys_path_vector(sys_path.size());
222 size_t i = 0;
223 for (auto item : sys_path) {
224 sys_path_vector[i++] = py::str(item);
225 }
226 return std::make_tuple(sys_path, sys_path_vector);
