41  bool mpiSyncStatusAnd(bool status, int mpi_rank, int mpi_num_procs, const std::string &taskDesc) {
42  // Expect 0 is good and 1 is no good for goodCode
43  // TODO: assert this in constructor or somewhere, or maybe just in a unit test
44  unsigned short codeBuffer;
45  bool printMessage = !taskDesc.empty();
46  // For the other ranks, start by properly setting the status code value in the buffer and send to rank 0
47  if (mpi_rank != 0) {
48  codeBuffer = status ? MPI_HF_SUB_CODE_GOOD : MPI_HF_SUB_CODE_BAD;
49  MPI_Send(&codeBuffer, 1, MPI_UNSIGNED_SHORT, 0, 0, MPI_COMM_WORLD);
50  }
51  // In rank 0, the first step is to receive and process codes from the other ranks into unified global status
52  else {
53  for (int i = 1; i < mpi_num_procs; ++i) {
54  MPI_Recv(&codeBuffer, 1, MPI_UNSIGNED_SHORT, i, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
55  // If any is ever "not good", overwrite status to be "false"
56  if (codeBuffer != MPI_HF_SUB_CODE_GOOD) {
57  if (printMessage) {
58  std::cout << "Rank " << i << " not successful/ready after " << taskDesc << std::endl;
59  }
60  status = false;
61  }
62  }
63  // Rank 0 must also now prepare the codeBuffer value for broadcasting the global status
64  codeBuffer = status ? MPI_HF_SUB_CODE_GOOD : MPI_HF_SUB_CODE_BAD;
65  }
66  // Execute broadcast of global status rooted at rank 0
67  MPI_Bcast(&codeBuffer, mpi_num_procs - 1, MPI_UNSIGNED_SHORT, 0, MPI_COMM_WORLD);
68  return codeBuffer == MPI_HF_SUB_CODE_GOOD;
69  }
